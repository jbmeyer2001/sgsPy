63template <typename T>
64void
65findMinMax(
66 helper::RasterBandMetaData& band,
67 int width,
68 int yStartBlock,
69 int yEndBlock,
70 T& min,
71 T& max,
72 std::mutex& mutex,
73 std::condition_variable& cv,
74 bool& finished)
75{
76 int xBlocks = (width + band.xBlockSize - 1) / band.xBlockSize;
77
78 min = std::numeric_limits<T>::max();
79 max = std::numeric_limits<T>::min();
80 T nan = static_cast<T>(band.nan);
81 void *p_data = VSIMalloc3(band.xBlockSize, band.yBlockSize, band.size);
82
83 //calculate raster band minimum and maximum values
84 for (int yBlock = yStartBlock; yBlock < yEndBlock; yBlock++) {
85 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
86 int xValid, yValid;
87 band.p_mutex->lock();
88 band.p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
89 band.p_mutex->unlock();
90
91 //read the band into memory
92 rasterBandIO(band, p_data, band.xBlockSize, band.yBlockSize, xBlock, yBlock, xValid, yValid, true, true);
93
94 for (int y = 0; y < yValid; y++) {
95 int index = y * width;
96 for (int x = 0; x < xValid; x++) {
97 T val = reinterpret_cast<T *>(p_data)[index];
98 if (val != nan && !std::isnan(val)) {
99 min = std::min(min, val);
100 max = std::max(max, val);
101 }
102 index++;
103 }
104 }
105 }
106 }
107
108 VSIFree(p_data);
109
110 mutex.lock();
111 finished = true;
112 mutex.unlock();
113 cv.notify_all();
114}

144template <typename T>
145void
146setBins(
147 T min,
148 T max,
149 int nBins,
150 GDALDataType type,
151 std::vector<double>& dbins,
152 std::vector<T>& tbins)
153{
154 dbins.resize(nBins + 1);
155 tbins.resize(nBins);
156
157 //determine bin values as doubles to display to the user.
158 double step = ((double)max - (double)min) / ((double)nBins);
159 double cur = (double)min;
160 for (int i = 0; i <= nBins; i++) {
161 dbins[i] = cur;
162 cur += step;
163 }
164
165 //set bin values as the data type used so less data type conversion is necessary in later code
166 if (type == GDT_Float32 || type == GDT_Float64) {
167 for (int i = 0; i < nBins; i++) {
168 tbins[i] = static_cast<T>(dbins[i]);
169 }
170 }
171 else {
172 tbins[0] = min;
173 cur += step;
174 for (int i = 1; i < nBins; i++) {
175 tbins[i] = std::ceil(cur);
176 cur += step;
177 }
178 }
179}

225template <typename T>
226void
227populationDistribution(
228 helper::RasterBandMetaData& band,
229 int width,
230 int yBlockStart,
231 int yBlockEnd,
232 int nBins,
233 std::vector<T>& binVals,
234 std::vector<int64_t>& counts,
235 std::mutex& mutex,
236 std::condition_variable& cv,
237 bool& finished)
238{
239 int xBlocks = (width + band.xBlockSize - 1) / band.xBlockSize;
240
241 T nan = static_cast<T>(band.nan);
242 void *p_data = VSIMalloc3(band.xBlockSize, band.yBlockSize, band.size);
243
244 std::vector<int64_t> chunkCounts(counts.size(), 0);
245 for (int yBlock = yBlockStart; yBlock < yBlockEnd; yBlock++) {
246 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
247 int xValid, yValid;
248 band.p_mutex->lock();
249 band.p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
250 band.p_mutex->unlock();
251
252 //read the band into memory
253 rasterBandIO(band, p_data, band.xBlockSize, band.yBlockSize, xBlock, yBlock, xValid, yValid, true, true);
254
255 for (int y = 0; y < yValid; y++) {
256 int index = y * width;
257 for (int x = 0; x < xValid; x++) {
258 T val = reinterpret_cast<T *>(p_data)[index];
259
260 if (val != nan && !std::isnan(val)) {
261 for (int i = nBins - 1; i >= 0; i--) {
262 if (binVals[i] <= val) {
263 chunkCounts[i]++;
264 break;
265 }
266 }
267 }
268 index++;
269 }
270 }
271 }
272 }
273
274 VSIFree(p_data);
275
276 mutex.lock();
277 finished = true;
278 for (int i = 0; i < nBins; i++) {
279 counts[i] += chunkCounts[i];
280 }
281 mutex.unlock();
282 cv.notify_all();
283}

298template <typename T>
299std::vector<int64_t>
300sampleDistribution(
301 helper::RasterBandMetaData& band,
302 std::vector<helper::Index>& samples,
303 std::vector<T> binVals,
304 int nBins)
305{
306 std::vector<int64_t> retval(nBins, 0);
307
308 T val;
309 for (const helper::Index& index : samples) {
310 band.p_band->RasterIO(GF_Read, index.x, index.y, 1, 1, &val, 1, 1, band.type, 0, 0);
311 for (int i = nBins - 1; i >= 0; i--) {
312 if (binVals[i] <= val) {
313 retval[i]++;
314 break;
315 }
316 }
317 }
318
319 return retval;
320}

348template <typename T>
349void
350calculateDist(
351 helper::RasterBandMetaData& band,
352 std::vector<helper::Index>& sampled,
353 int height,
354 int width,
355 int nBins,
356 std::unordered_map<std::string, std::pair<std::vector<double>, std::vector<int64_t>>>& retval,
357 int nThreads)
358{
359 pybind11::gil_scoped_acquire acquire;
360 boost::asio::thread_pool pool(nThreads);
361
362 //determine number of chunks and chunks size for each thread. A chunk is a group of blocks.
363 int yBlocks = (height + band.yBlockSize - 1) / band.yBlockSize;
364 int chunksize = std::max(1, yBlocks / nThreads);
365 int nChunks = (yBlocks + chunksize - 1) / chunksize;
366
367 std::mutex mutex;
368 std::condition_variable cv;
369
370 //don't use std::vector<bool> because can't reference individual bools from it
371 bool *chunksFinished = reinterpret_cast<bool *>(VSIMalloc2(nChunks, sizeof(bool)));
372
373 //determine min and max values in the raster
374 std::vector<T> mins(nChunks);
375 std::vector<T> maxs(nChunks);
376 for (int chunk = 0; chunk < nChunks; chunk++) {
377 int yStartBlock = chunk * chunksize;
378 int yEndBlock = std::min(yStartBlock + chunksize, yBlocks);
379 chunksFinished[chunk] = false;
380
381 boost::asio::post(pool, std::bind(findMinMax<T>,
382 std::ref(band),
383 width,
384 yStartBlock,
385 yEndBlock,
386 std::ref(mins[chunk]),
387 std::ref(maxs[chunk]),
388 std::ref(mutex),
389 std::ref(cv),
390 std::ref(chunksFinished[chunk])
391 ));
392 }
393
394 //using cv, wait for all of the threads to finish calculating their chunk before
395 //calculating the final min and max values
396 std::unique_lock lock(mutex);
397 bool allChunksFinished = true;
398 for (int chunk = 0; chunk < nChunks; chunk++) {
399 allChunksFinished &= chunksFinished[chunk];
400 }
401
402 while (!allChunksFinished) {
403 cv.wait(lock);
404 allChunksFinished = true;
405 for (int chunk = 0; chunk < nChunks; chunk++) {
406 allChunksFinished &= chunksFinished[chunk];
407 }
408 }
409
410 //calculate the final min/max values
411 T min = *std::min_element(mins.begin(), mins.end());
412 T max = *std::max_element(maxs.begin(), maxs.end());
413
414 //call the setBins function which sets the vector<double> bins to return to the user,
415 //and the vector<T> bins to use to create the distribution. There is a seperate vector<T>
416 //bins object so that while iterating through every pixel, they don't have to be cast to
417 //type double to check.
418 std::vector<double> dbins;
419 std::vector<T> tbins;
420 setBins<T>(min, max, nBins, band.type, dbins, tbins);
421
422 //determine the population distribution of the raster band
423 std::vector<int64_t> counts(nBins, 0);
424 for (int chunk = 0; chunk < nChunks; chunk++) {
425 int yStartBlock = chunk * chunksize;
426 int yEndBlock = std::min(yStartBlock + chunksize, yBlocks);
427
428 chunksFinished[chunk] = false;
429
430 boost::asio::post(pool, std::bind(populationDistribution<T>,
431 std::ref(band),
432 width,
433 yStartBlock,
434 yEndBlock,
435 nBins,
436 std::ref(tbins),
437 std::ref(counts),
438 std::ref(mutex),
439 std::ref(cv),
440 std::ref(chunksFinished[chunk])
441 ));
442 }
443
444 //using cv, wait for all of the threads to finish calculating their chunk before
445 //calculating the final counts in each bin
446 allChunksFinished = true;
447 for (int chunk = 0; chunk < nChunks; chunk++) {
448 allChunksFinished &= chunksFinished[chunk];
449 }
450
451 while (!allChunksFinished) {
452 cv.wait(lock);
453 allChunksFinished = true;
454 for (int chunk = 0; chunk < nChunks; chunk++) {
455 allChunksFinished &= chunksFinished[chunk];
456 }
457 }
458
459 VSIFree(chunksFinished);
460
461 retval.insert({std::string("population"), {dbins, counts}});
462
463 if (sampled.size() != 0) {
464 std::vector<int64_t> sampleCounts = sampleDistribution<T>(band, sampled, tbins, nBins);
465 retval.insert({std::string("sample"), {dbins, sampleCounts}});
466 }
467
468 pybind11::gil_scoped_release release;
469}

494std::unordered_map<std::string, std::pair<std::vector<double>, std::vector<int64_t>>>
495dist(
496 raster::GDALRasterWrapper *p_raster,
497 int index,
498 vector::GDALVectorWrapper *p_vector,
499 std::string layer,
500 int nBins,
501 int nThreads)
502{
503 std::mutex datasetMutex;
504 double *GT = p_raster->getGeotransform();
505 double IGT[6];
506 GDALInvGeoTransform(GT, IGT);
507
508 //get samples as vector if indices, if samples are given
509 std::vector<helper::Index> sampled;
510 if (p_vector) {
511 OGRLayer *p_layer = p_vector->getDataset()->GetLayerByName(layer.c_str());
512 for (const auto& p_feature : *p_layer) {
513 OGRGeometry *p_geometry = p_feature->GetGeometryRef();
514 switch (wkbFlatten(p_geometry->getGeometryType())) {
515 case OGRwkbGeometryType::wkbPoint: {
516 OGRPoint *p_point = p_geometry->toPoint();
517 double xCoord = p_point->getX();
518 double yCoord = p_point->getY();
519 sampled.push_back(helper::Index(
520 static_cast<int>(IGT[0] + xCoord * IGT[1] + yCoord * IGT[2]),
521 static_cast<int>(IGT[3] + xCoord * IGT[4] + yCoord * IGT[5])
522 ));
523 break;
524 }
525 case OGRwkbGeometryType::wkbMultiPoint: {
526 for (const auto& p_point : *p_geometry->toMultiPoint()) {
527 double xCoord = p_point->getX();
528 double yCoord = p_point->getY();
529 sampled.push_back(helper::Index(
530 static_cast<int>(IGT[0] + xCoord * IGT[1] + yCoord * IGT[2]),
531 static_cast<int>(IGT[3] + xCoord * IGT[4] + yCoord * IGT[5])
532 ));
533 }
534 break;
535 }
536 default:
537 throw std::runtime_error("encountered a geometry which was not a Point MultiPoint.");
538 }
539 }
540 }
541
542 std::mutex bandMutex;
543
544 helper::RasterBandMetaData band;
545 band.p_band = p_raster->getRasterBand(index);
546 band.type = p_raster->getRasterBandType(index);
547 band.size = p_raster->getRasterBandTypeSize(index);
548 band.p_mutex = &datasetMutex;
549 band.nan = band.p_band->GetNoDataValue();
550 band.p_band->GetBlockSize(&band.xBlockSize, &band.yBlockSize);
551 band.p_mutex = &bandMutex;
552
553 int height = p_raster->getHeight();
554 int width = p_raster->getWidth();
555
556 std::unordered_map<std::string, std::pair<std::vector<double>, std::vector<int64_t>>> retval;
557 switch (band.type) {
558 case GDT_Int8:
559 calculateDist<int8_t>(band, sampled, height, width, nBins, retval, nThreads);
560 break;
561 case GDT_UInt16:
562 calculateDist<uint16_t>(band, sampled, height, width, nBins, retval, nThreads);
563 break;
564 case GDT_Int16:
565 calculateDist<int16_t>(band, sampled, height, width, nBins, retval, nThreads);
566 break;
567 case GDT_UInt32:
568 calculateDist<uint32_t>(band, sampled, height, width, nBins, retval, nThreads);
569 break;
570 case GDT_Int32:
571 calculateDist<int32_t>(band, sampled, height, width, nBins, retval, nThreads);
572 break;
573 case GDT_Float32:
574 calculateDist<float>(band, sampled, height, width, nBins, retval, nThreads);
575 break;
576 case GDT_Float64:
577 calculateDist<double>(band, sampled, height, width, nBins, retval, nThreads);
578 break;
579 default:
580 throw std::runtime_error("raster pixel data type not supported.");
581 }
582
583 return retval;
584}
