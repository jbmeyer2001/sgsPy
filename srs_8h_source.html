49template <typename T>
50inline void
51processBlock(
52 helper::RasterBandMetaData& band,
53 access::Access& access,
54 existing::Existing& existing,
55 std::vector<helper::Index>& indices,
56 helper::RandValController& rand,
57 int xBlock,
58 int yBlock,
59 int xValid,
60 int yValid)
61{
62 T nan = static_cast<T>(band.nan);
63 int8_t *p_access = reinterpret_cast<int8_t *>(access.band.p_buffer);
64
65 for (int y = 0; y < yValid; y++) {
66 size_t blockIndex = static_cast<size_t>(y * band.xBlockSize);
67 for (int x = 0; x < xValid; x++) {
68 //get val
69 T val = helper::getPixelValueDependingOnType<T>(band.type, band.p_buffer, blockIndex);
70
71 //check nan
72 bool isNan = val == nan || std::isnan(val);
73 if (isNan) {
74 blockIndex++;
75 continue;
76 }
77
78 //check access
79 if (access.used && p_access[blockIndex] == 1) {
80 blockIndex++;
81 continue;
82 }
83
84 helper::Index index = {x + xBlock * band.xBlockSize, y + yBlock * band.yBlockSize};
85
86 //check existing
87 if (existing.used && existing.containsIndex(index.x, index.y)) {
88 blockIndex++;
89 continue;
90 }
91
92 //check rand val
93 if (!rand.next()) {
94 blockIndex++;
95 continue;
96 }
97
98 //add index to indices
99 indices.push_back(index);
100
101 //increment within-block index
102 blockIndex++;
103 }
104 }
105}

173std::tuple<std::vector<std::vector<double>>, vector::GDALVectorWrapper *, size_t>
174srs(
175 raster::GDALRasterWrapper *p_raster,
176 size_t numSamples,
177 double mindist,
178 vector::GDALVectorWrapper *p_existing,
179 vector::GDALVectorWrapper *p_access,
180 std::string layerName,
181 double buffInner,
182 double buffOuter,
183 bool plot,
184 std::string tempFolder,
185 std::string filename)
186{
187 GDALAllRegister();
188
189 int width = p_raster->getWidth();
190 int height = p_raster->getHeight();
191 double *GT = p_raster->getGeotransform();
192 bool useMindist = mindist != 0;
193 std::mutex mutex;
194
195 std::vector<double> xCoords, yCoords;
196 std::vector<size_t> indexes;
197
198 //step 3: get first raster band
199 helper::RasterBandMetaData band;
200 band.p_band = p_raster->getRasterBand(0);
201 band.type = p_raster->getRasterBandType(0);
202 band.size = p_raster->getRasterBandTypeSize(0);
203 band.nan = band.p_band->GetNoDataValue();
204 band.p_mutex = &mutex;
205 band.p_band->GetBlockSize(&band.xBlockSize, &band.yBlockSize);
206 band.p_buffer = VSIMalloc3(band.xBlockSize, band.yBlockSize, band.size);
207
208 //create output dataset before doing anything which will take a long time in case of failure.
209 GDALDriver *p_driver = GetGDALDriverManager()->GetDriverByName("MEM");
210 if (!p_driver) {
211 throw std::runtime_error("unable to create output sample dataset driver.");
212 }
213 GDALDataset *p_samples = p_driver->Create("", 0, 0, 0, GDT_Unknown, nullptr);
214 if (!p_samples) {
215 throw std::runtime_error("unable to create output dataset with driver.");
216 }
217
218 vector::GDALVectorWrapper *p_wrapper = new vector::GDALVectorWrapper(p_samples, std::string(p_raster->getDataset()->GetProjectionRef()));
219 OGRLayer *p_layer = p_samples->CreateLayer("samples", p_wrapper->getSRS(), wkbPoint, nullptr);
220 if (!p_layer) {
221 throw std::runtime_error("unable to create output dataset layer.");
222 }
223
224 //generate access structure
225 access::Access access(
226 p_access,
227 p_raster,
228 layerName,
229 buffInner,
230 buffOuter,
231 true,
232 tempFolder,
233 band.xBlockSize,
234 band.yBlockSize
235 );
236
237 if (access.used) {
238 access.band.p_buffer = VSIMalloc3(band.xBlockSize, band.yBlockSize, access.band.size);
239 }
240
241 //generate existing structure
242 existing::Existing existing(
243 p_existing,
244 GT,
245 p_raster->getWidth(),
246 p_layer,
247 plot,
248 xCoords,
249 yCoords
250 );
251
252 int xBlocks = (width + band.xBlockSize - 1) / band.xBlockSize;
253 int yBlocks = (height + band.yBlockSize - 1) / band.yBlockSize;
254
255 std::vector<bool> randVals(band.xBlockSize * band.yBlockSize);
256 int randValIndex = band.xBlockSize * band.yBlockSize;
257
258 //fast random number generator using xoshiro256++
259 //https://vigna.di.unimi.it/ftp/papers/ScrambledLinear.pdf
260 xso::xoshiro_4x64_plus rng;
261
262 //the multiplier which will be multiplied by the 53 most significant bits of the output of the
263 //random number generator to see whether a pixel should be added or not. The multiplier is
264 //a uint64_t number where the least significant n bits are 1 and the remaining are 0. The pixel
265 //is added when the least significant n bits (of the bit shifted 53 bits) within the rng match
266 //those of the multiplier. The probability a pixel is add is then (1/2^n). Using this method,
267 //knowing the amount of pixels, estimating the number of nan pixels, taking into account mindist
268 //and accessible area, we can estimate a percentage chance for each pixel and set up a multiplier
269 //to make that percentage happen. Doing this enables retaining only a small portion of pixel data
270 //and reducing memory footprint significantly, otherwise the index of every pixel
271 //would have to be stored, which would not be feasible for large rasters.
272 uint64_t multiplier = helper::getProbabilityMultiplier(
273 width,
274 height,
275 p_raster->getPixelWidth(),
276 p_raster->getPixelHeight(),
277 4,
278 numSamples,
279 useMindist,
280 access.area
281 );
282
283 helper::RandValController rand(band.xBlockSize, band.yBlockSize, multiplier, &rng);
284
285 std::vector<helper::Index> indices;
286 for (int yBlock = 0; yBlock < yBlocks; yBlock++) {
287 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
288 int xValid, yValid;
289
290 //read block
291 band.p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
292 helper::rasterBandIO(band, band.p_buffer, band.xBlockSize, band.yBlockSize, xBlock, yBlock, xValid, yValid, true, false);
293 //read = true
294 //read access block if necessary
295 if (access.used) {
296 helper::rasterBandIO(access.band, access.band.p_buffer, band.xBlockSize, band.yBlockSize, xBlock, yBlock, xValid, yValid, true, false);
297 }
298
299 //calculate random values
300 rand.calculateRandValues();
301
302 //process block
303 switch (band.type) {
304 case GDT_Int8:
305 processBlock<int8_t>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
306 break;
307 case GDT_UInt16:
308 processBlock<uint16_t>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
309 break;
310 case GDT_Int16:
311 processBlock<int16_t>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
312 break;
313 case GDT_UInt32:
314 processBlock<uint32_t>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
315 break;
316 case GDT_Int32:
317 processBlock<int32_t>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
318 break;
319 case GDT_Float32:
320 processBlock<float>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
321 break;
322 case GDT_Float64:
323 processBlock<double>(band, access, existing, indices, rand, xBlock, yBlock, xValid, yValid);
324 break;
325 default:
326 throw std::runtime_error("raster pixel data type is not supported.");
327 }
328 }
329 }
330
331 std::shuffle(indices.begin(), indices.end(), rng);
332
333 size_t samplesAdded = existing.used ? existing.count() : 0;
334 size_t i = 0;
335 helper::NeighborMap neighbor_map;
336 double mindist_sq = mindist * mindist;
337
338 while (samplesAdded < numSamples && i < indices.size()) {
339 helper::Index index = indices[i];
340 bool valid = true;
341
342 double x = GT[0] + index.x * GT[1] + index.y * GT[2];
343 double y = GT[3] + index.x * GT[4] + index.y * GT[5];
344
345 if (useMindist) {
346 valid = helper::is_valid_sample(x, y, neighbor_map, mindist, mindist_sq);
347 }
348
349 if (valid) {
350 OGRPoint point = OGRPoint(x, y);
351 helper::addPoint(&point, p_layer);
352
353 samplesAdded++;
354
355 if (plot) {
356 xCoords.push_back(x);
357 yCoords.push_back(y);
358 }
359 }
360 i++;
361 }
362
363 if (filename != "") {
364 try {
365 p_wrapper->write(filename);
366 }
367 catch (const std::exception& e) {
368 std::cout << "Exception thrown trying to write file: " << e.what() << std::endl;
369 }
370 }
371
372 //free allocated memory
373 VSIFree(band.p_buffer);
374 if (access.used) {
375 VSIFree(access.band.p_buffer);
376 }
377
378 return {{xCoords, yCoords}, p_wrapper, samplesAdded};
379}
