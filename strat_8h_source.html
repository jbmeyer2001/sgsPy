40std::vector<int64_t>
41calculateAllocation(
42 int64_t numSamples,
43 std::string allocation,
44 std::vector<int64_t> strataCounts,
45 std::vector<double> weights,
46 int64_t numPixels)
47{
48 std::vector<int64_t> retval;
49 int64_t remainder = numSamples;
50 int64_t numStrata = strataCounts.size();
51 if (allocation == "prop") {
52 //allocate the samples per stratum according to stratum size
53 int64_t pixelsPerSample = numPixels / numSamples;
54
55 //add 1 if pixelsPerSample was truncated down, to avoid adding too many samples
56 pixelsPerSample += static_cast<int64_t>(pixelsPerSample * numSamples < numPixels);
57
58 for (int64_t i = 0; i < numStrata; i++) {
59 int64_t count = strataCounts[i] / pixelsPerSample;
60 retval.push_back(count);
61 remainder -= count;
62 }
63 }
64 else if (allocation == "equal") {
65 //determine the count of samples per strata
66 int64_t strataSampleCount = numSamples / numStrata;
67
68 for (int64_t i = 0; i < numStrata; i++) {
69 retval.push_back(strataSampleCount);
70 remainder -= strataSampleCount;
71 }
72 }
73 else if (allocation == "manual" || allocation == "optim") {
74 //allocate samples accordign to weights.
75 //Optim allocation calculates these weights whereas when using manual the weights are given by the user
76 for (int64_t i = 0; i < numStrata; i++) {
77 int64_t count = static_cast<int64_t>(static_cast<double>(numSamples) * weights[i]);
78 retval.push_back(count);
79 remainder -= count;
80 }
81 }
82 else {
83 throw std::runtime_error("allocation method must be one of 'prop', 'equal', 'manual', or 'optim'.");
84 }
85
86 //redistribute remainder pixels among strata, and check strata sizes
87 for (int64_t i = numStrata; i > 0; i--) {
88 int64_t extra = remainder / i;
89 retval[i - 1] += extra;
90 remainder -= extra;
91
92 if (retval[i - 1] > strataCounts[i - 1]) {
93 std::cout << "warning: strata " << i - 1 << " does not have enough pixels for the full " << retval[i - 1] << " samples it should recieve. There will be less than " << numSamples << " final samples." << std::endl;
94 retval[i - 1] = strataCounts[i - 1];
95 }
96 }
97
98 return retval;
99}

109struct OptimAllocationDataManager {
110 helper::RasterBandMetaData band;
111 std::vector<helper::Variance> variances;

113
123 OptimAllocationDataManager(raster::GDALRasterWrapper *p_raster, int bandNum, std::string allocation) {
124 if (!p_raster || allocation != "optim") {
125 return;
126 }
127
128 this->band.p_band = p_raster->getRasterBand(bandNum);
129 this->band.type = p_raster->getRasterBandType(bandNum);
130 this->band.size = p_raster->getRasterBandTypeSize(bandNum);
131 this->band.p_buffer = nullptr; //allocated later
132 this->band.nan = this->band.p_band->GetNoDataValue();
133 this->band.p_band->GetBlockSize(&this->band.xBlockSize, &this->band.yBlockSize);
134 this->used = true;
135 }

141 ~OptimAllocationDataManager() {
142 if (this->used) {
143 VSIFree(this->band.p_buffer);

156 inline void
157 init(int numStrata, int xBlockSize, int yBlockSize) {
158 this->variances.resize(numStrata);

173 inline void

186 inline void

204 inline std::vector<double>
205 getAllocationPercentages(void) {
206 std::vector<double> retval(variances.size());
207
208 double total = 0;
209 for (size_t i = 0; i < variances.size(); i++) {
210 double stdev = variances[i].getStdev();
211 double count = static_cast<double>(variances[i].getCount());
212 double product = count == 0 ? 0 : stdev * count; //if count == 0 stdev will be nan, this stops the nan from spreading
213 retval[i] = product;
214 total += product;
215 }
216
217 for (size_t i = 0; i < variances.size(); i++) {
218 retval[i] = retval[i] / total;
219 }
220
221 return retval;
222 }

242class IndexStorageVectors {
243private:
244 std::vector<int64_t> strataCounts;
245
246 std::vector<int64_t> indexCountPerStrata;
247 std::vector<std::vector<helper::Index>> indexesPerStrata;
248
249 std::vector<int64_t> firstXIndexCountPerStrata;
250 std::vector<std::vector<helper::Index>> firstXIndexesPerStrata;
251
252 int64_t numStrata;
253 int64_t x;
254
255public:
263 IndexStorageVectors(int64_t numStrata, int64_t x) {
264 this->numStrata = numStrata;
265 this->x = x;
266
267 this->strataCounts.resize(numStrata);
268
269 this->indexCountPerStrata.resize(numStrata, 0);
270 this->indexesPerStrata.resize(numStrata);
271
272 this->firstXIndexCountPerStrata.resize(numStrata, 0);
273 this->firstXIndexesPerStrata.resize(numStrata);
274 for (int64_t i = 0; i < numStrata; i++) {
275 this->firstXIndexesPerStrata[i].resize(x);
276 }
277 }
278
284 inline void
285 updateStrataCounts(int strata) {
286 this->strataCounts[strata]++;
287 }
288
296 inline void
297 updateIndexesVector(int strata, helper::Index& index) {
298 this->indexesPerStrata[strata].push_back(index);
299 this->indexCountPerStrata[strata]++;
300 }
301
309 inline void
310 updateFirstXIndexesVector(int strata, helper::Index& index) {
311 int64_t i = firstXIndexCountPerStrata[strata];
312 if (i < this->x) {
313 firstXIndexesPerStrata[strata][i] = index;
314 firstXIndexCountPerStrata[strata]++;
315 }
316 else if (i == this->x) {
317 std::vector<helper::Index>().swap(firstXIndexesPerStrata[strata]);
318 firstXIndexCountPerStrata[strata]++;
319 }
320 }
321
340 inline std::vector<std::vector<helper::Index> *>
341 getStrataIndexVectors(std::vector<int64_t> existing, std::vector<int64_t> strataSampleCounts, xso::xoshiro_4x64_plus& rng) {
342 std::vector<std::vector<helper::Index> *> retval(numStrata);
343
344 for (int64_t i = 0; i < numStrata; i++) {
345 int64_t existingSamples = existing[i];
346 int64_t desiredSamples = strataSampleCounts[i];
347 int64_t remainingSamples = desiredSamples - existingSamples;
348
349 int64_t probIndexesCount = indexCountPerStrata[i];
350 int64_t firstXIndexesCount = firstXIndexCountPerStrata[i];
351
352 if (probIndexesCount >= remainingSamples || firstXIndexesCount > x) {
353 auto begin = this->indexesPerStrata[i].begin();
354 auto end = this->indexesPerStrata[i].end();
355 std::shuffle(begin, end, rng);
356 retval[i] = &this->indexesPerStrata[i];
357 }
358 else {
359 this->firstXIndexesPerStrata[i].resize(firstXIndexesCount);
360 auto begin = this->firstXIndexesPerStrata[i].begin();
361 auto end = this->firstXIndexesPerStrata[i].end();
362 std::shuffle(begin, end, rng);
363 retval[i] = &this->firstXIndexesPerStrata[i];
364 }
365 }
366
367 return retval;
368 }

376 getStrataCounts(void) {
377 return this->strataCounts;
378 }

385 inline int64_t
386 getNumDataPixels(void) {
387 int64_t retval = 0;
388
389 for (const int64_t& count : this->strataCounts) {
390 retval += count;
391 }
392
393 return retval;
394 }

438template <typename T>
439std::vector<int64_t>
440processBlocksStratRandom(
441 int numSamples,
442 int numStrata,
443 helper::RasterBandMetaData& band,
444 access::Access& access,
445 existing::Existing& existing,
446 IndexStorageVectors& indices,
447 std::vector<std::vector<OGRPoint>>& existingSamples,
448 uint64_t multiplier,
449 xso::xoshiro_4x64_plus& rng,
450 std::string allocation,
451 OptimAllocationDataManager& optim,
452 std::vector<double> weights,
453 int width,
454 int height)
455{
456 T nanInt = static_cast<T>(band.nan);
457 int xBlockSize = band.xBlockSize;
458 int yBlockSize = band.yBlockSize;
459
460 int xBlocks = (width + xBlockSize - 1) / xBlockSize;
461 int yBlocks = (height + yBlockSize - 1) / yBlockSize;
462
463 band.p_buffer = VSIMalloc3(xBlockSize, yBlockSize, band.size);
464 T *p_buffer = reinterpret_cast<T *>(band.p_buffer);
465 int8_t *p_access = nullptr;
466 if (access.used) {
467 access.band.p_buffer = VSIMalloc3(xBlockSize, yBlockSize, access.band.size);
468 p_access = reinterpret_cast<int8_t *>(access.band.p_buffer);
469 }
470
471 helper::RandValController rand(band.xBlockSize, band.yBlockSize, multiplier, &rng);
472
473 if (optim.used) {
474 optim.init(numStrata, xBlockSize, yBlockSize);
475 }
476
477 for (int yBlock = 0; yBlock < yBlocks; yBlock++) {
478 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
479 int xValid, yValid;
480
481 //read block
482 band.p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
483 helper::rasterBandIO(band, band.p_buffer, xBlockSize, yBlockSize, xBlock, yBlock, xValid, yValid, true, false);
484
485 //read access block
486 if (access.used) {
487 helper::rasterBandIO(access.band, access.band.p_buffer, xBlockSize, yBlockSize, xBlock, yBlock, xValid, yValid, true, false);
488 }
489
490 //read mraster block for optim
491 if (optim.used) {
492 optim.readNewBlock(xBlockSize, yBlockSize, xBlock, yBlock, xValid, yValid);
493 }
494
495 //calculate rand vals
496 rand.calculateRandValues();
497
498 //iterate through block and update vectors
499 for (int y = 0; y < yValid; y++) {
500 int blockIndex = y * xBlockSize;
501 for (int x = 0; x < xValid; x++) {
502 T val = p_buffer[blockIndex];
503 helper::Index index = {x + xBlock * xBlockSize, y + yBlock * yBlockSize};
504
505 bool isNan = val == nanInt;
506 bool accessible = !access.used || p_access[blockIndex] != 1;
507 bool alreadySampled = existing.used && existing.containsIndex(index.x, index.y);
508
509 //check nan
510 if (isNan) {
511 blockIndex++;
512 continue;
513 }
514
515 //update optim allocation variance calculations
516 if (optim.used) {
517 optim.update(blockIndex, val);
518 }
519
520 //udpate strata counts
521 indices.updateStrataCounts(val);
522
523 //update existing sampled strata
524 if (alreadySampled) {
525 existingSamples[val].push_back(existing.getPoint(index.x, index.y));
526 }
527
528 //add val to stored indices
529 if (accessible && !alreadySampled) {
530 indices.updateFirstXIndexesVector(val, index);
531
532 if (rand.next()) {
533 indices.updateIndexesVector(val, index);
534 }
535 }
536
537 //increment block index
538 blockIndex++;
539 }
540 }
541 }
542 }
543
544 VSIFree(band.p_buffer);
545 if (access.used) {
546 VSIFree(access.band.p_buffer);
547 }
548
549 if (optim.used) {
550 weights = optim.getAllocationPercentages();
551 }
552
553 std::vector<int64_t> strataSampleCounts = calculateAllocation(
554 numSamples,
555 allocation,
556 indices.getStrataCounts(),
557 weights,
558 indices.getNumDataPixels()
559 );
560
561 return strataSampleCounts;
562}

595struct FocalWindow {
596 int wrow, wcol;
597 int width;
598 int vpad, hpad;
599 std::vector<bool> m;
600 std::vector<bool> valid;
601
611 FocalWindow(int wrow, int wcol, int width) {
612 this->wrow = wrow;
613 this->vpad = wrow / 2;
614 this->wcol = wcol;
615 this->hpad = wcol / 2;
616 this->width = width;
617 this->m.resize(wrow * width, false);
618 this->valid.resize(wrow * width, false);
619 }
620
629 inline void
630 reset(int row) {
631 int start = (row % wrow) * this->width;
632 int end = start + this->width;
633 for (int i = start; i < end; i++) {
634 m[i] = false;
635 valid[i] = false;
636 }
637 }
638
661 inline bool
662 check(int x, int y) {
663 y = y % wrow;
664 switch (wrow) {
665 case 3:
666 return this->m[x] &&
667 this->m[x + width] &&
668 this->m[x + width * 2] &&
669 this->valid[x + y * width];
670 case 5:
671 return this->m[x] &&
672 this->m[x + width] &&
673 this->m[x + width * 2] &&
674 this->m[x + width * 3] &&
675 this->m[x + width * 4] &&
676 this->valid[x + y * width];
677 case 7:
678 return this->m[x] &&
679 this->m[x + width] &&
680 this->m[x + width * 2] &&
681 this->m[x + width * 3] &&
682 this->m[x + width * 4] &&
683 this->m[x + width * 5] &&
684 this->m[x + width * 6] &&
685 this->valid[x + y * width];
686 default:
687 throw std::runtime_error("wrow must be one of 3, 5, 7.");
688 }
689 }
690};

764template <typename T>
765std::vector<int64_t>
766processBlocksStratQueinnec(
767 int numSamples,
768 int numStrata,
769 helper::RasterBandMetaData& band,
770 access::Access& access,
771 existing::Existing& existing,
772 IndexStorageVectors& indices,
773 IndexStorageVectors& queinnecIndices,
774 FocalWindow& fw,
775 std::vector<std::vector<OGRPoint>>& existingSamples,
776 uint64_t multiplier,
777 uint64_t queinnecMultiplier,
778 xso::xoshiro_4x64_plus& rng,
779 std::string allocation,
780 OptimAllocationDataManager& optim,
781 std::vector<double> weights,
782 int width,
783 int height)
784{
785 T nanInt = static_cast<T>(band.nan);
786
787 //adjust blocks to be a large chunk of scanlines
788 int xBlockSize = band.xBlockSize;
789 int yBlockSize = band.yBlockSize;
790 if (xBlockSize != width) {
791 xBlockSize = width;
792 }
793 else {
794 yBlockSize = std::min(128, height);
795 }
796
797 //allocate required memory
798 band.p_buffer = VSIMalloc3(xBlockSize, yBlockSize + fw.vpad * 2, band.size);
799 T *p_buffer = reinterpret_cast<T *>(band.p_buffer);
800 int8_t *p_access = nullptr;
801 if (access.used) {
802 access.band.p_buffer = VSIMalloc3(xBlockSize, yBlockSize, access.band.size);
803 p_access = reinterpret_cast<int8_t *>(access.band.p_buffer);
804 }
805
806 helper::RandValController rand(xBlockSize, yBlockSize, multiplier, &rng);
807 helper::RandValController queinnecRand(xBlockSize, yBlockSize, queinnecMultiplier, &rng);
808
809 if (optim.used) {
810 optim.init(numStrata, xBlockSize, yBlockSize);
811 }
812
813 //get number of blocks
814 int yBlocks = (height + yBlockSize - 1) / yBlockSize;
815
816 int fwyi = 0;
817 for (int yBlock = 0; yBlock < yBlocks; yBlock++) {
818 //read block
819 int xOff = 0;
820 int yOff = yBlock * yBlockSize;
821 int xValid = width;
822 int yValid = std::min(yBlockSize, height - yBlock * yBlockSize);
823
824 //read block
825 if (yBlock == 0) {
826 CPLErr err = band.p_band->RasterIO(GF_Read, xOff, yOff, xValid, yValid, band.p_buffer, xValid, yValid, band.type, 0, 0);
827 if (err) {
828 throw std::runtime_error("error reading block from raster.");
829 }
830 }
831 else {
832 int stratYOff = yOff - fw.vpad * 2;
833 int stratYValid = yValid + fw.vpad * 2;
834 CPLErr err = band.p_band->RasterIO(GF_Read, xOff, stratYOff, xValid, stratYValid, band.p_buffer, xValid, stratYValid, band.type, 0, 0);
835 if (err) {
836 throw std::runtime_error("error reading block from raster.");
837 }
838 }
839
840 //read access block
841 if (access.used) {
842 CPLErr err = access.band.p_band->RasterIO(GF_Read, xOff, yOff, xValid, yValid, access.band.p_buffer, xValid, yValid, GDT_Int8, 0, 0);
843 if (err) {
844 throw std::runtime_error("error reading block from raster.");
845 }
846 }
847
848 //read mraster block for optim
849 if (optim.used) {
850 CPLErr err = optim.band.p_band->RasterIO(GF_Read, xOff, yOff, xValid, yValid, optim.band.p_buffer, xValid, yValid, optim.band.type, 0, 0);
851 if (err) {
852 throw std::runtime_error("error reading block from raster.");
853 }
854 }
855
856 //calculate rand vals
857 rand.calculateRandValues();
858 queinnecRand.calculateRandValues();
859
860 //calculate the within-block index. In the case where there is a padding at the top of the block,
861 //adjust for that.
862 int newBlockStart = (yBlock == 0) ? 0 : width * fw.vpad * 2;
863
864 //iterate through block and update vectors
865 for (int y = 0; y < yValid; y++) {
866 //reset upcomming row in focal window matrix
867 fw.reset(yBlock * yBlockSize + y);
868
869 //the indexes from 0 to the horizontal pad cannot be a queinnec index
870 //because they are too close to the edges of the raster
871 for (int x = 0; x < fw.hpad; x++) {
872 T val = p_buffer[newBlockStart + y * width + x];
873 helper::Index index = {x, y + yBlock * yBlockSize};
874
875 bool isNan = val == nanInt;
876 bool accessible = !access.used || p_access[y * width + x] != 1;
877 bool alreadySampled = existing.used && existing.containsIndex(index.x, index.y);
878
879 //check nan
880 if (isNan) {
881 continue;
882 }
883
884 //update optim allocation variance calculations
885 if (optim.used) {
886 optim.update(y * width + x, val);
887 }
888
889 //update strata counts
890 indices.updateStrataCounts(val);
891
892 //update existing samled strata
893 if (alreadySampled) {
894 existingSamples[val].push_back(existing.getPoint(index.x, index.y));
895 }
896
897 //add val to stored indices
898 if (accessible && !alreadySampled) {
899 indices.updateFirstXIndexesVector(val, index);
900
901 if (rand.next()) {
902 indices.updateIndexesVector(val, index);
903 }
904 }
905 }
906
907 for (int x = fw.hpad; x < width - fw.hpad; x++) {
908 T val = p_buffer[newBlockStart + y * width + x];
909 helper::Index index = {x, y + yBlock * yBlockSize};
910
911 bool isNan = val == nanInt;
912 bool accessible = !access.used || p_access[y * width + x] != 1;
913 bool alreadySampled = existing.used && existing.containsIndex(index.x, index.y);
914
915 //check nan
916 if (isNan) {
917 continue;
918 }
919
920 //update optim allocation variance calculations
921 if (optim.used) {
922 optim.update(y * width + x, val);
923 }
924
925 //update strata counts
926 indices.updateStrataCounts(val);
927
928 //update exisitng sampled strata
929 if (alreadySampled) {
930 existingSamples[val].push_back(existing.getPoint(index.x, index.y));
931 };
932
933 //set focal window matrix value by checking horizontal indices within focal window
934 int start = newBlockStart + y * width + x - fw.hpad;
935 switch (fw.wcol) {
936 case 3:
937 fw.m[fwyi + x] = p_buffer[start] == p_buffer[start + 1] &&
938 p_buffer[start] == p_buffer[start + 2];
939 break;
940 case 5:
941 fw.m[fwyi + x] = p_buffer[start] == p_buffer[start + 1] &&
942 p_buffer[start] == p_buffer[start + 2] &&
943 p_buffer[start] == p_buffer[start + 3] &&
944 p_buffer[start] == p_buffer[start + 4];
945 break;
946 case 7:
947 fw.m[fwyi + x] = p_buffer[start] == p_buffer[start + 1] &&
948 p_buffer[start] == p_buffer[start + 2] &&
949 p_buffer[start] == p_buffer[start + 3] &&
950 p_buffer[start] == p_buffer[start + 4] &&
951 p_buffer[start] == p_buffer[start + 5] &&
952 p_buffer[start] == p_buffer[start + 6];
953 break;
954 default:
955 throw std::runtime_error("wcol must be one of 3, 5, 6.");
956 }
957
958 //add val to stored indices and update focal window validity
959 if (accessible && !alreadySampled) {
960 indices.updateFirstXIndexesVector(val, index);
961
962 if (rand.next()) {
963 indices.updateIndexesVector(val, index);
964 }
965
966 fw.valid[fwyi + x] = true;
967 }
968
969 //add index to queinnec indices if surrounding focal window is all the same
970 helper::Index fwIndex = {x, index.y - fw.vpad};
971 if (fw.check(fwIndex.x, fwIndex.y)) {
972 bool add = false;
973 int start = newBlockStart + y * width + x - 2 * width * fw.vpad;
974 switch (fw.wrow) {
975 case 3:
976 add = p_buffer[start] == p_buffer[start + width * 1] &&
977 p_buffer[start] == p_buffer[start + width * 2];
978 break;
979 case 5:
980 add = p_buffer[start] == p_buffer[start + width * 1] &&
981 p_buffer[start] == p_buffer[start + width * 2] &&
982 p_buffer[start] == p_buffer[start + width * 3] &&
983 p_buffer[start] == p_buffer[start + width * 4];
984 break;
985 case 7:
986 add = p_buffer[start] == p_buffer[start + width * 1] &&
987 p_buffer[start] == p_buffer[start + width * 2] &&
988 p_buffer[start] == p_buffer[start + width * 3] &&
989 p_buffer[start] == p_buffer[start + width * 4] &&
990 p_buffer[start] == p_buffer[start + width * 5] &&
991 p_buffer[start] == p_buffer[start + width * 6];
992 break;
993 }
994
995 if (add) {
996 //(we know if we've made it here that val is the same for both the fw add and the current index)
997 queinnecIndices.updateFirstXIndexesVector(val, fwIndex);
998
999 if (queinnecRand.next()) {
1000 queinnecIndices.updateIndexesVector(val, fwIndex);
1001 }
1002 }
1003 }
1004 }
1005
1006 //the indexes from width - horizontal pad to width cannot be a queinnec index
1007 //because they are too close to the edges of the raster
1008 for (int x = width - fw.hpad; x < width; x++) {
1009 T val = p_buffer[newBlockStart + y * width + x];
1010 helper::Index index = {x, y + yBlock * yBlockSize};
1011
1012 bool isNan = val == nanInt;
1013 bool accessible = !access.used || p_access[y * width + x] != 1;
1014 bool alreadySampled = existing.used && existing.containsIndex(index.x, index.y);
1015
1016 //check nan
1017 if (isNan) {
1018 continue;
1019 }
1020
