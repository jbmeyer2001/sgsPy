171inline T
172getPixelValueDependingOnType(
173 GDALDataType type,
174 void *p_data,
175 size_t index
176) {
177 switch (type) {
178 case GDT_Int8:
179 return static_cast<T>(((int8_t *)p_data)[index]);
180 case GDT_UInt16:
181 return static_cast<T>(((uint16_t *)p_data)[index]);
182 case GDT_Int16:
183 return static_cast<T>(((int16_t *)p_data)[index]);
184 case GDT_UInt32:
185 return static_cast<T>(((uint32_t *)p_data)[index]);
186 case GDT_Int32:
187 return static_cast<T>(((int32_t *)p_data)[index]);
188 case GDT_Float32:
189 return static_cast<T>(((float *)p_data)[index]);
190 case GDT_Float64:
191 return static_cast<T>(((double *)p_data)[index]);
192 default:
193 throw std::runtime_error("raster pixel data type not supported.");
194 }

373inline GDALDataset *
374createDataset(
375 std::string filename,
376 std::string driverName,
377 int width,
378 int height,
379 double *geotransform,
380 std::string projection,
381 RasterBandMetaData *bands,
382 size_t bandCount,
383 bool useTiles,
384 std::map<std::string, std::string>& driverOptions)
385{
386 GDALDriver *p_driver = GetGDALDriverManager()->GetDriverByName(driverName.c_str());
387 if (!p_driver) {
388 throw std::runtime_error("unable to find dataset driver.");
389 }
390
391 char **papszOptions = nullptr;
392 if (useTiles) {
393 if (driverOptions.find("TILED") == driverOptions.end()) {
394 papszOptions = CSLSetNameValue(papszOptions, "TILED", "YES");
395 }
396 if (driverOptions.find("BLOCKXSIZE") == driverOptions.end()) {
397 const char *xBlockSizeOption = std::to_string(bands[0].xBlockSize).c_str();
398 papszOptions = CSLSetNameValue(papszOptions, "BLOCKXSIZE", xBlockSizeOption);
399 }
400 if (driverOptions.find("BLOCKYSIZE") == driverOptions.end()) {
401 const char *yBlockSizeOption = std::to_string(bands[0].yBlockSize).c_str();
402 papszOptions = CSLSetNameValue(papszOptions, "BLOCKYSIZE", yBlockSizeOption);
403 }
404 }
405
406 for (auto const& [key, val] : driverOptions) {
407 papszOptions = CSLSetNameValue(papszOptions, key.c_str(), val.c_str());
408 }
409
410 GDALDataset *p_dataset = p_driver->Create(
411 filename.c_str(),
412 width,
413 height,
414 bandCount,
415 bands[0].type,
416 papszOptions
417 );
418 CSLDestroy(papszOptions);
419
420 if (!p_dataset) {
421 throw std::runtime_error("unable to create dataset with driver.");
422 }
423
424 CPLErr err = p_dataset->SetGeoTransform(geotransform);
425 if (err) {
426 throw std::runtime_error("error setting geotransform.");
427 }
428
429 err = p_dataset->SetProjection(projection.c_str());
430 if (err) {
431 throw std::runtime_error("error setting projection.");
432 }
433
434 for (size_t band = 0; band < bandCount; band++) {
435 GDALRasterBand *p_band = p_dataset->GetRasterBand(band + 1);
436 p_band->SetDescription(bands[band].name.c_str());
437 p_band->SetNoDataValue(bands[band].nan);
438 p_band->GetBlockSize(&bands[band].xBlockSize, &bands[band].yBlockSize);
439 bands[band].p_band = p_band;
440 }
441
442 return p_dataset;
443}

641inline void
642rasterBandIO(
643 RasterBandMetaData& band,
644 void *p_buffer,
645 int xBlockSize,
646 int yBlockSize,
647 int xBlock,
648 int yBlock,
649 int xValid,
650 int yValid,
651 bool read,
652 bool threaded = true)
653{
654 CPLErr err;
655 bool useBlock = xBlockSize == band.xBlockSize &&
656 yBlockSize == band.yBlockSize;
657
658 if (threaded) {
659 band.p_mutex->lock();
660 }
661 if (useBlock && read) {
662 err = read ?
663 band.p_band->ReadBlock(xBlock, yBlock, p_buffer) :
664 band.p_band->WriteBlock(xBlock, yBlock, p_buffer);
665 }
666 else {
667 err = band.p_band->RasterIO(
668 read ? GF_Read : GF_Write,
669 xBlock * xBlockSize,
670 yBlock * yBlockSize,
671 xValid,
672 yValid,
673 p_buffer,
674 xBlockSize,
675 yBlockSize,
676 band.type,
677 0,
678 0
679 );
680 }
681 if (threaded) {
682 band.p_mutex->unlock();
683 }
684 if (err) {
685 throw read ?
686 std::runtime_error("unable to read block from raster.") :
687 std::runtime_error("unable to write block to raster.");
688 }

831class Variance {
832 private:
833 int64_t k = 0; //count
834 double M = 0; //running mean
835 double S = 0; //sum of squares
836 double oldM = 0;
837
838 public:
839 inline void
845 update(double x) {
846 k++;
847 oldM = M;
848
849 //update running mean
850 M = M + (x - M) / static_cast<double>(k);
851
852 //update sum of squares
853 S = S + (x - M) * (x - oldM);
854 }
855
861 inline double
862 getMean() {
863 return M;
864 }
865
872 inline double
873 getStdev() {
874 double variance = S / static_cast<double>(k);
875 return std::sqrt(variance);
876 }
877
883 inline int64_t
884 getCount() {
885 return this->k;
886 }
887};

938inline uint64_t
939getProbabilityMultiplier(double width, double height, double pixelWidth, double pixelHeight, int startMult, int numSamples, bool useMindist, double accessibleArea) {
940 double numer = static_cast<double>(numSamples * startMult * (useMindist ? 3 : 1));
941 double denom = height * width;
942
943 if (accessibleArea != -1) {
944 double totalArea = width * pixelWidth * height * pixelHeight;
945 numer *= (totalArea / accessibleArea);
946 }
947
948 if (numer > denom) {
949 return 0;
950 }
951
952 uint8_t bits = static_cast<uint8_t>(std::ceil(std::log2(denom) - std::log2(numer)));
953 return (1 << bits) - 1;
954}

975class RandValController {
976private:
977 std::vector<bool> randVals;
978 size_t randValIndex = 0;
979 uint64_t multiplier = 0;
980 xso::xoshiro_4x64_plus *p_rng = nullptr;
981 bool alwaysTrue = false;
982
983public:
993 RandValController(int xBlockSize, int yBlockSize, uint64_t multiplier, xso::xoshiro_4x64_plus *p_rng) {
994 if (this->multiplier == 0) {
995 this->alwaysTrue = true;
996 }
997 else {
998 this->randVals.resize(xBlockSize * yBlockSize);
999 this->randValIndex = static_cast<size_t>(xBlockSize * yBlockSize);
1000 this->multiplier = multiplier;
1001 this->p_rng = p_rng;
1002 }
1003 }
1004
1017 inline void
1018 calculateRandValues(void) {
1019 if (alwaysTrue) {
1020 return;
1021 }
1022
1023 for (size_t i = 0; i < randValIndex; i++) {
1024 randVals[i] = (((*p_rng)() >> 11) & multiplier) == multiplier;
1025 }
1026 randValIndex = 0;
1027 }
1028
1032 inline bool
1033 next(void) {
1034 if (alwaysTrue) {
1035 return true;
1036 }
1037
1038 bool retval = randVals[randValIndex];
1039 randValIndex++;
1040 return retval;
1041 }
1042};

1071inline bool is_valid_sample(double x, double y, NeighborMap& neighbor_map, float mindist, float mindist_sq) {
1072 int cx = static_cast<int>(std::floor(x / mindist));
1073 int cy = static_cast<int>(std::floor(y / mindist));
1074
1075 for (int dx = -1; dx <= 1; dx++) {
1076 for (int dy = -1; dy <= 1; dy++) {
1077 std::pair<int, int> neighbor = {cx + dx, cy + dy};
1078
1079 auto it = neighbor_map.find(neighbor);
1080 if (it == neighbor_map.end())
1081 continue;
1082
1083 for (const auto &[nx, ny] : it->second) {
1084 float dxp = x - nx;
1085 float dyp = y - ny;
1086 float dist_sq = dxp * dxp + dyp * dyp;
1087
1088 if (dist_sq < mindist_sq) {
1089 return false;
1090 }
1091 }
1092 }
1093 }
1094
1095 // add point to neighborhood
1096 neighbor_map[{cx, cy}].emplace_back(x, y);
1097 return true;
1098}
