167) {
168 switch (type) {

179 case GDT_Float32:

181 case GDT_Float64:
182 return static_cast<T>(((double *)p_data)[index]);
183 default:
184 throw std::runtime_error("raster pixel data type not supported.");

364inline GDALDataset *
365createDataset(
366 std::string filename,
367 std::string driverName,
368 int width,
369 int height,
370 double *geotransform,
371 std::string projection,
372 RasterBandMetaData *bands,
373 size_t bandCount,
374 bool useTiles,
375 std::map<std::string, std::string>& driverOptions)
376{
377 GDALDriver *p_driver = GetGDALDriverManager()->GetDriverByName(driverName.c_str());
378 if (!p_driver) {
379 throw std::runtime_error("unable to find dataset driver.");
380 }
381
382 char **papszOptions = nullptr;
383 if (useTiles) {
384 if (driverOptions.find("TILED") == driverOptions.end()) {
385 papszOptions = CSLSetNameValue(papszOptions, "TILED", "YES");
386 }
387 if (driverOptions.find("BLOCKXSIZE") == driverOptions.end()) {
388 const char *xBlockSizeOption = std::to_string(bands[0].xBlockSize).c_str();
389 papszOptions = CSLSetNameValue(papszOptions, "BLOCKXSIZE", xBlockSizeOption);
390 }
391 if (driverOptions.find("BLOCKYSIZE") == driverOptions.end()) {
392 const char *yBlockSizeOption = std::to_string(bands[0].yBlockSize).c_str();
393 papszOptions = CSLSetNameValue(papszOptions, "BLOCKYSIZE", yBlockSizeOption);
394 }
395 }
396
397 for (auto const& [key, val] : driverOptions) {
398 papszOptions = CSLSetNameValue(papszOptions, key.c_str(), val.c_str());
399 }
400
401 GDALDataset *p_dataset = p_driver->Create(
402 filename.c_str(),
403 width,
404 height,
405 bandCount,
406 bands[0].type,
407 papszOptions
408 );
409 CSLDestroy(papszOptions);
410
411 if (!p_dataset) {
412 throw std::runtime_error("unable to create dataset with driver.");
413 }
414
415 CPLErr err = p_dataset->SetGeoTransform(geotransform);
416 if (err) {
417 throw std::runtime_error("error setting geotransform.");
418 }
419
420 err = p_dataset->SetProjection(projection.c_str());
421 if (err) {
422 throw std::runtime_error("error setting projection.");
423 }
424
425 for (size_t band = 0; band < bandCount; band++) {
426 GDALRasterBand *p_band = p_dataset->GetRasterBand(band + 1);
427 p_band->SetDescription(bands[band].name.c_str());
428 p_band->SetNoDataValue(bands[band].nan);
429 p_band->GetBlockSize(&bands[band].xBlockSize, &bands[band].yBlockSize);
430 bands[band].p_band = p_band;
431 }
432
433 return p_dataset;
434}

632inline void
633rasterBandIO(
634 RasterBandMetaData& band,
635 void *p_buffer,
636 int xBlockSize,
637 int yBlockSize,
638 int xBlock,
639 int yBlock,
640 int xValid,
641 int yValid,
642 bool read,
643 bool threaded = true)
644{
645 CPLErr err;
646 bool useBlock = xBlockSize == band.xBlockSize &&
647 yBlockSize == band.yBlockSize;
648
649 if (threaded) {
650 band.p_mutex->lock();
651 }
652 if (useBlock && read) {
653 err = read ?
654 band.p_band->ReadBlock(xBlock, yBlock, p_buffer) :
655 band.p_band->WriteBlock(xBlock, yBlock, p_buffer);
656 }
657 else {
658 err = band.p_band->RasterIO(
659 read ? GF_Read : GF_Write,
660 xBlock * xBlockSize,
661 yBlock * yBlockSize,
662 xValid,
663 yValid,
664 p_buffer,
665 xBlockSize,
666 yBlockSize,
667 band.type,
668 0,
669 0
670 );
671 }
672 if (threaded) {
673 band.p_mutex->unlock();
674 }
675 if (err) {
676 throw read ?
677 std::runtime_error("unable to read block from raster.") :
678 std::runtime_error("unable to write block to raster.");
679 }

747class Variance {
748 private:
749 int64_t k = 0; //count
750 double M = 0; //running mean
751 double S = 0; //sum of squares
752 double oldM = 0;
753
754 public:
755 inline void
761 update(double x) {
762 k++;
763 oldM = M;
764
765 //update running mean
766 M = M + (x - M) / static_cast<double>(k);
767
768 //update sum of squares
769 S = S + (x - M) * (x - oldM);
770 }
771
777 inline double
778 getMean() {
779 return M;
780 }
781
788 inline double
789 getStdev() {
790 double variance = S / static_cast<double>(k);
791 return std::sqrt(variance);
792 }
793
799 inline int64_t
800 getCount() {
801 return this->k;
802 }
803};

854inline uint64_t
855getProbabilityMultiplier(double width, double height, double pixelWidth, double pixelHeight, int startMult, int numSamples, bool useMindist, double accessibleArea) {
856 double numer = static_cast<double>(numSamples * startMult * (useMindist ? 3 : 1));
857 double denom = height * width;
858
859 if (accessibleArea != -1) {
860 double totalArea = width * pixelWidth * height * pixelHeight;
861 numer *= (totalArea / accessibleArea);
862 }
863
864 if (numer > denom) {
865 return 0;
866 }
867
868 uint8_t bits = static_cast<uint8_t>(std::ceil(std::log2(denom) - std::log2(numer)));
869 return (1 << bits) - 1;
870}

891class RandValController {
892private:
893 std::vector<bool> randVals;
894 size_t randValIndex = 0;
895 uint64_t multiplier = 0;
896 xso::xoshiro_4x64_plus *p_rng = nullptr;
897 bool alwaysTrue = false;
898
899public:
909 RandValController(int xBlockSize, int yBlockSize, uint64_t multiplier, xso::xoshiro_4x64_plus *p_rng) {
910 if (this->multiplier == 0) {
911 this->alwaysTrue = true;
912 }
913 else {
914 this->randVals.resize(xBlockSize * yBlockSize);
915 this->randValIndex = static_cast<size_t>(xBlockSize * yBlockSize);
916 this->multiplier = multiplier;
917 this->p_rng = p_rng;
918 }
919 }

933 inline void
934 calculateRandValues(void) {
935 if (alwaysTrue) {
936 return;
937 }
938
939 for (size_t i = 0; i < randValIndex; i++) {
940 randVals[i] = (((*p_rng)() >> 11) & multiplier) == multiplier;
941 }
942 randValIndex = 0;
943 }
944
948 inline bool
949 next(void) {
950 if (alwaysTrue) {
951 return true;
952 }
953
954 bool retval = randVals[randValIndex];
955 randValIndex++;
956 return retval;
957 }

987inline bool is_valid_sample(double x, double y, NeighborMap& neighbor_map, float mindist, float mindist_sq) {
988 int cx = static_cast<int>(std::floor(x / mindist));
989 int cy = static_cast<int>(std::floor(y / mindist));
990
991 for (int dx = -1; dx <= 1; dx++) {
992 for (int dy = -1; dy <= 1; dy++) {
993 std::pair<int, int> neighbor = {cx + dx, cy + dy};
994
995 auto it = neighbor_map.find(neighbor);
996 if (it == neighbor_map.end())
997 continue;
998
999 for (const auto &[nx, ny] : it->second) {
1000 float dxp = x - nx;
1001 float dyp = y - ny;
1002 float dist_sq = dxp * dxp + dyp * dyp;
1003
1004 if (dist_sq < mindist_sq) {
1005 return false;
1006 }
1007 }
1008 }
1009 }
1010
1011 // add point to neighborhood
1012 neighbor_map[{cx, cy}].emplace_back(x, y);
1013 return true;
1014}
