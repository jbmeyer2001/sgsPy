75 if (!p_vect) {

80 this->width = width;
81
82 std::vector<std::string> layerNames = p_vect->getLayerNames();
83 if (layerNames.size() > 1) {
84 throw std::runtime_error("the file containing existing sample points must have only a single layer.");
85 }
86
87 //invert geotransform so we can use IGT to convert from point to indexes
88 GDALInvGeoTransform(GT, this->IGT);
89
90 std::string name = layerNames[0];
91 OGRLayer *p_layer = p_vect->getLayer(name);
92
93 for (const auto& p_feature : *p_layer) {
94 OGRGeometry *p_geometry = p_feature->GetGeometryRef();
95 switch (wkbFlatten(p_geometry->getGeometryType())) {
96 case OGRwkbGeometryType::wkbPoint: {
97 OGRPoint *p_point = p_geometry->toPoint();
98 int64_t index = helper::point2index<int64_t>(p_point->getX(), p_point->getY(), IGT, width);
99 this->samples.emplace(index, *p_point);
100 if (p_samples) {
101 helper::addPoint(p_point, p_samples);
102 if (plot) {
103 xCoords.push_back(p_point->getX());
104 yCoords.push_back(p_point->getY());
105 }
106 }
107 break;
108 }
109 case OGRwkbGeometryType::wkbMultiPoint: {
110 for (const auto& p_point : *p_geometry->toMultiPoint()) {
111 int64_t index = helper::point2index<int64_t>(p_point->getX(), p_point->getY(), IGT, width);
112 this->samples.emplace(index, *p_point);
113 if (p_samples) {
114 helper::addPoint(p_point, p_samples);
115 if (plot) {
116 xCoords.push_back(p_point->getX());
117 yCoords.push_back(p_point->getY());
118 }
119 }
120 }
121 break;
122 }
123 default:
124 throw std::runtime_error("the file containing existing sample points must have only Point or MultiPoint geometries.");
125 }
126 }
127
128 this->used = true;
129 }
