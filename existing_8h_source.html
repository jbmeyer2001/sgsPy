83 OGRFieldDefn existingField("existing", OFTInteger);
84 OGRErr err = p_samples->CreateField(&existingField);
85 if (err) {
86 throw std::runtime_error("cannot create 'existing' field in output layer.");
87 }
88
89 this->width = width;
90
91 std::vector<std::string> layerNames = p_vect->getLayerNames();
92 if (layerNames.size() > 1) {
93 throw std::runtime_error("the file containing existing sample points must have only a single layer.");
94 }
95
96 //check to ensure spatial reference system of raster and vector match
97 std::string rastProj = p_rast->getDataset()->GetProjectionRef();
98 OGRSpatialReference rastSRS;
99 rastSRS.importFromWkt(rastProj.c_str());
100 OGRSpatialReference *p_sampSRS = p_samples->GetSpatialRef();
101 if (!rastSRS.IsSame(p_sampSRS)) {
102 throw std::runtime_error("existing sample vector and raster do not have the same spatial reference system.");
103 }
104
105 //invert geotransform so we can use IGT to convert from point to indexes
106 GDALInvGeoTransform(GT, this->IGT);
107
108 std::string name = layerNames[0];
109 OGRLayer *p_layer = p_vect->getLayer(name);
110 helper::Field fieldExistingTrue("existing", 1);
111
112 for (const auto& p_feature : *p_layer) {
113 OGRGeometry *p_geometry = p_feature->GetGeometryRef();
114 switch (wkbFlatten(p_geometry->getGeometryType())) {
115 case OGRwkbGeometryType::wkbPoint: {
116 OGRPoint *p_point = p_geometry->toPoint();
117 int64_t index = helper::point2index<int64_t>(p_point->getX(), p_point->getY(), IGT, width);
118 this->samples.emplace(index, *p_point);
119 if (addAllPoints) {
120 helper::addPoint(p_point, p_samples, &fieldExistingTrue);
121 if (plot) {
122 xCoords.push_back(p_point->getX());
123 yCoords.push_back(p_point->getY());
124 }
125 }
126 break;
127 }
128 case OGRwkbGeometryType::wkbMultiPoint: {
129 for (const auto& p_point : *p_geometry->toMultiPoint()) {
130 int64_t index = helper::point2index<int64_t>(p_point->getX(), p_point->getY(), IGT, width);
131 this->samples.emplace(index, *p_point);
132 if (addAllPoints) {
133 helper::addPoint(p_point, p_samples, &fieldExistingTrue);
134 if (plot) {
135 xCoords.push_back(p_point->getX());
136 yCoords.push_back(p_point->getY());
137 }
138 }
139 }
140 break;
141 }
142 default:
143 throw std::runtime_error("the file containing existing sample points must have only Point or MultiPoint geometries.");
144 }
145 }
146
147 this->used = true;
148 }
