55template <typename T>
56inline size_t
57getQuantile(T val, std::vector<T>& quantiles) {
58 auto it = std::lower_bound(quantiles.begin(), quantiles.end(), val);
59 return (it == quantiles.end()) ?
60 quantiles.size() :
61 std::distance(quantiles.begin(), it);
62}

72template <typename T>
73class CLHSDataManager {
74 private:
75 std::vector<T> features;
76 std::vector<int> x;
77 std::vector<int> y;
78 size_t fi; //features index
79 size_t count;
80 int64_t size;
81 uint64_t ucount;
82
83 std::vector<std::vector<T>> corr;
84
85 int nFeat;
86 int nSamp;
87
88 xso::xoshiro_4x64_plus *p_rng = nullptr;
89 uint64_t mask = 0;
90
91 public:
101 CLHSDataManager(int nFeat, int nSamp, xso::xoshiro_4x64_plus *p_rng) {
102 this->nFeat = nFeat;
103 this->nSamp = nSamp;
104 this->count = 0;
105 this->fi = 0;
106 this->size = MILLION;
107 this->features.resize(MILLION * nFeat);
108 this->x.resize(MILLION);
109 this->y.resize(MILLION);
110
111 this->p_rng = p_rng;
112 }
113
126 inline void
127 addPoint(T *p_features, int x, int y) {
128 for (int f = 0; f < nFeat; f++) {
129 features[fi] = p_features[f];
130 fi++;
131 }
132
133 this->x[this->count] = x;
134 this->y[this->count] = y;
135 this->count++;
136
137 if (this->count == this->size) {
138 this->features.resize(this->features.size() + MILLION * this->nFeat);
139 this->x.resize(this->x.size() + MILLION);
140 this->y.resize(this->y.size() + MILLION);
141 this->size += MILLION;
142 }
143 }
144
165 inline void
166 finalize(std::vector<std::vector<T>>& corr) {
167 if (this->count < static_cast<size_t>(this->nSamp)) {
168 throw std::runtime_error("not enough points saved during raster iteration to conduct clhs sampling.");
169 }
170
171 this->corr = corr;
172
173 this->x.resize(this->size);
174 this->y.resize(this->size);
175 this->features.resize(this->count * nFeat);
176 this->ucount = static_cast<uint64_t>(this->count);
177
178 //use bit twiddling to fill the mask
179 this->mask = static_cast<uint64_t>(this->count);
180 this->mask |= this->mask >> 1;
181 this->mask |= this->mask >> 2;
182 this->mask |= this->mask >> 4;
183 this->mask |= this->mask >> 8;
184 this->mask |= this->mask >> 16;
185 this->mask |= this->mask >> 32;
186 }
187
202 inline uint64_t
203 randomIndex() {
204 uint64_t index = ((*p_rng)() >> 11) & mask;
205
206 while (index > ucount) {
207 index = ((*p_rng)() >> 11) & mask;
208 }
209
210 return index;
211 }
212
219 inline void
220 getPoint(Point<T>& point, uint64_t index) {
221 point.p_features = this->features.data() + (index * nFeat);
222 point.x = x[index];
223 point.y = y[index];
224 }
225
240 inline T
241 quantileObjectiveFunc(std::vector<std::vector<int>>& sampleCountPerQuantile) {
242 int retval = 0;
243
244 for (const std::vector<int>& quantiles : sampleCountPerQuantile) {
245 for (const int& count : quantiles) {
246 retval += std::abs(count - 1);
247 }
248 }
249
250 return static_cast<T>(retval);
251 }
252
266 inline T
267 correlationObjectiveFunc(std::vector<std::vector<T>>& corr) {
268 T retval = 0;
269
270 for (size_t i = 0; i < this->corr.size(); i++) {
271 for (size_t j = 0; j < this->corr[i].size(); j++) {
272 retval += std::abs(corr[i][j] - this->corr[i][j]);
273 }
274 }
275
276 return retval;
277 }
278};

343template <typename T>
344inline void
345readRaster(
346 std::vector<helper::RasterBandMetaData>& bands,
347 CLHSDataManager<T>& clhs,
348 access::Access& access,
349 helper::RandValController& rand,
350 GDALDataType type,
351 std::vector<std::vector<T>>& quantiles,
352 size_t size,
353 int width,
354 int height,
355 int count,
356 int nSamp)
357{
358 std::vector<std::vector<T>> probabilities(count);
359 quantiles.resize(count);
360
361 for (int i = 0; i < count; i++) {
362 probabilities[i].resize(nSamp - 1);
363 quantiles[i].resize(nSamp - 1);
364
365 for (int j = 0; j < nSamp - 1; j++) {
366 probabilities[i][j] = static_cast<T>(j + 1) / static_cast<T>(nSamp);
367 }
368 }
369
370 int xBlockSize = bands[0].xBlockSize;
371 int yBlockSize = bands[0].yBlockSize;
372
373 int xBlocks = (width + xBlockSize - 1) / xBlockSize;
374 int yBlocks = (height + yBlockSize - 1) / yBlockSize;
375
376 double deps = .001;
377 float seps = .001;
378
379 std::vector<T> corrBuffer(count * xBlockSize * yBlockSize);
380 std::vector<std::vector<T>> quantileBuffers(count);
381 for (int i = 0; i < count; i++) {
382 quantileBuffers[i].resize(xBlockSize * yBlockSize);
383 }
384
385 if (access.used) {
386 access.band.p_buffer = VSIMalloc3(xBlockSize, yBlockSize, access.band.size);
387 }
388
389 //create descriptor for correlation matrix streaming calculation with oneDAL
390 const auto cor_desc = oneapi::dal::covariance::descriptor{}.set_result_options(oneapi::dal::covariance::result_options::cor_matrix);
391 oneapi::dal::covariance::partial_compute_result<> partial_result;
392
393 //create tasks for quantile streaming calculation with MKL
394 std::vector<VSLSSTaskPtr> quantileTasks(count);
395 int status;
396 MKL_INT quant_order_n = quantiles[0].size();
397 MKL_INT p = 1;
398 MKL_INT n = xBlockSize * yBlockSize;
399 MKL_INT nparams = VSL_SS_SQUANTS_ZW_PARAMS_N;
400 MKL_INT xstorage = VSL_SS_MATRIX_STORAGE_ROWS;
401
402 //MKL functions have different versions for single/double precision floating point data
403 if (type == GDT_Float64) {
404 for (int i = 0; i < count; i++) {
405 //reinterpret cast the pointer for compiler reasons
406 status = vsldSSNewTask(
407 &quantileTasks[i],
408 &p,
409 &n,
410 &xstorage,
411 reinterpret_cast<double *>(quantileBuffers[i].data()),
412 0,
413 0
414 );
415 }
416 }
417 else {
418 for (int i = 0; i < count; i++) {
419 //reinterpret cast the pointer for compiler reasons
420 status = vslsSSNewTask(
421 &quantileTasks[i],
422 &p,
423 &n,
424 &xstorage,
425 reinterpret_cast<float *>(quantileBuffers[i].data()),
426 0,
427 0
428 );
429 }
430 }
431
432 int8_t *p_access = reinterpret_cast<int8_t *>(access.band.p_buffer);
433
434 bool calledEditStreamQuantiles = false;
435 void *p_data = reinterpret_cast<void *>(corrBuffer.data());
436
437 for (int yBlock = 0; yBlock < yBlocks; yBlock++) {
438 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
439 //get block size
440 int xValid, yValid;
441 bands[0].p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
442
443 //read bands into memory
444 for (int i = 0; i < count ; i++) {
445 CPLErr err = bands[i].p_band->RasterIO(
446 GF_Read,
447 xBlock * xBlockSize,
448 yBlock * yBlockSize,
449 xValid,
450 yValid,
451 (void *)((size_t)p_data + i * size),
452 xValid,
453 yValid,
454 type,
455 size * static_cast<size_t>(count),
456 size * static_cast<size_t>(count) * static_cast<size_t>(xBlockSize)
457 );
458
459 if (err) {
460 throw std::runtime_error("Error reading data from raster band.");
461 }
462 }
463
464 //calculate rand vals
465 rand.calculateRandValues();
466
467 //read access band into memory if used
468 if (access.used) {
469 rasterBandIO(access.band, access.band.p_buffer, xBlockSize, yBlockSize, xBlock, yBlock, xValid, yValid, true, false);
470 }
471
472 //iterate through pixels
473 n = 0;
474 for (int y = 0; y < yValid; y++) {
475 int index = y * xBlockSize;
476 for (int x = 0; x < xValid; x++) {
477 bool isNan = false;
478 T *p_buff = corrBuffer.data() + (index * count);
479 for (int b = 0; b < count; b++) {
480 T val = p_buff[b];
481 isNan = std::isnan(val) || val == bands[b].nan;
482
483 if (isNan) {
484 break;
485 }
486
487 quantileBuffers[b][n] = val;
488 corrBuffer[n * count + b] = val;
489 }
490
491 if (!isNan) {
492 n++;
493
494 bool accessible = !access.used || p_access[index] != 1;
495 if (accessible && rand.next()) {
496 clhs.addPoint(
497 p_buff,
498 xBlock * xBlockSize + x,
499 yBlock * yBlockSize + y
500 );
501 }
502 }
503 index++;
504 }
505 }
506
507 if (n == 0) {
508 continue;
509 }
510
511 //MKL functions have different versions for single/double precision floating point data
512 if (type == GDT_Float64) {
513 if (!calledEditStreamQuantiles) {
514 for (int i = 0; i < count; i++) {
515 //reinterpret cast the pointers for compiler reasons
516 status = vsldSSEditStreamQuantiles(
517 quantileTasks[i],
518 &quant_order_n,
519 reinterpret_cast<double *>(probabilities[i].data()),
520 reinterpret_cast<double *>(quantiles[i].data()),
521 &nparams,
522 &deps
523 );
524 }
525 calledEditStreamQuantiles = true;
526 }
527 for (int i = 0; i < count; i++) {
528 status = vsldSSCompute(
529 quantileTasks[i],
530 VSL_SS_STREAM_QUANTS,
531 VSL_SS_METHOD_SQUANTS_ZW_FAST
532 );
533 }
534 }
535 else { //type == GDT_Float32
536 if (!calledEditStreamQuantiles) {
537 for (int i = 0; i < count; i++) {
538 //reinterpret cast the pointers for compiler reasons
539 status = vslsSSEditStreamQuantiles(
540 quantileTasks[i],
541 &quant_order_n,
542 reinterpret_cast<float *>(probabilities[i].data()),
543 reinterpret_cast<float *>(quantiles[i].data()),
544 &nparams,
545 &seps
546 );
547 }
548 calledEditStreamQuantiles = true;
549 }
550 for (int i = 0; i < count; i++) {
551 status = vslsSSCompute(
552 quantileTasks[i],
553 VSL_SS_STREAM_QUANTS,
554 VSL_SS_METHOD_SQUANTS_ZW_FAST
555 );
556 }
557
558 }
559
560 //update correlation matrix calculations
561 DALHomogenTable table = DALHomogenTable(corrBuffer.data(), n, count, [](const T *){}, oneapi::dal::data_layout::row_major);
562 partial_result = oneapi::dal::partial_compute(cor_desc, partial_result, table);
563 }
564 }
565
566 if (access.used) {
567 VSIFree(access.band.p_buffer);
568 }
569
570 //calculate and update clhs data manager with correlation matrix
571 auto result = oneapi::dal::finalize_compute(cor_desc, partial_result);
572 auto correlation = result.get_cor_matrix();
573
574 oneapi::dal::row_accessor<const T> acc {correlation};
575
576 n = 0;
577 std::vector<std::vector<T>> corr(count);
578 for (int i = 0; i < count; i++) {
579 corr[i].resize(count);
580 auto row = acc.pull({i, i + 1});
581
582 for (int j = 0; j < count; j++) {
583 corr[i][j] = row[j];
584 }
585
586 status = (type == GDT_Float64) ?
587 vsldSSCompute(quantileTasks[i], VSL_SS_STREAM_QUANTS, VSL_SS_METHOD_SQUANTS_ZW) :
588 vslsSSCompute(quantileTasks[i], VSL_SS_STREAM_QUANTS, VSL_SS_METHOD_SQUANTS_ZW);
589 status = vslSSDeleteTask(&quantileTasks[i]);
590 }
591
592 clhs.finalize(corr);
593}

628template <typename T>
629inline void
630selectSamples(std::vector<std::vector<T>>& quantiles,
631 CLHSDataManager<T>& clhs,
632 xso::xoshiro_4x64_plus& rng,
633 int iterations,
634 int nSamp,
635 int nFeat,
636 OGRLayer *p_layer,
637 double *GT,
638 bool plot,
639 std::vector<double>& xCoords,
640 std::vector<double>& yCoords)
641{
642 std::uniform_real_distribution<T> dist(0.0, 1.0);
643 std::uniform_int_distribution<int> indexDist(0, nSamp - 1);
644
645 std::vector<std::vector<T>> corr(nFeat);
646 std::vector<std::vector<int>> sampleCountPerQuantile(nFeat);
647 std::vector<std::vector<std::unordered_set<uint64_t>>> samplesPerQuantile(nFeat);
648 for (int i = 0; i < nFeat; i++) {
649 sampleCountPerQuantile[i].resize(nSamp, 0);
650 samplesPerQuantile[i].resize(nSamp);
651 corr[i].resize(nFeat);
652 }
653
654 std::vector<T> features(nSamp * nFeat);
655 std::vector<int> x(nSamp);
656 std::vector<int> y(nSamp);
657
658 int featuresSize = nSamp * nFeat;
659
660 //indices vector used to get an index value in O(1) time after this vector is randomly indexed
661 std::vector<uint64_t> indices(nSamp);
662
663 //indices map used to check whether an index is already used in O(1) time, and to keep track of it's index in the 'indices' vector
664 std::unordered_map<uint64_t, int> indicesMap;
665
666 //get first random samples
667 int i = 0;
668 while (i < nSamp) {
669 uint64_t index = clhs.randomIndex();
670
671 if (!indicesMap.contains(index)) {
672 indicesMap.emplace(index, i);
673
674 Point<T> p;
675 clhs.getPoint(p, index);
676
677 indices[i] = index;
678 x[i] = p.x;
679 y[i] = p.y;
680
681 int fi = i * nFeat;
682 for (int f = 0; f < nFeat; f++) {
683 T val = p.p_features[f];
684 features[fi + f] = val;
685
686 int q = getQuantile<T>(val, quantiles[f]);
687 sampleCountPerQuantile[f][q]++;
688 samplesPerQuantile[f][q].insert(index);
689 }
690
691 i++;
692 }
693 }
694
695 //define covariance calculation
696 DALHomogenTable table = DALHomogenTable(features.data(), nSamp, nFeat, [](const T *){}, oneapi::dal::data_layout::row_major);
697 const auto cor_desc = oneapi::dal::covariance::descriptor{}.set_result_options(oneapi::dal::covariance::result_options::cor_matrix);
698 const auto result = oneapi::dal::compute(cor_desc, table);
699 oneapi::dal::row_accessor<const T> acc {result.get_cor_matrix()};
700 for (int i = 0; i < nFeat; i++) {
701 auto row = acc.pull({i, i + 1});
702
703 for (int j = 0; j < nFeat; j++) {
704 corr[i][j] = row[j];
705 }
706 }
707
708 double temp = 1;
709 double d = temp / static_cast<double>(iterations);
710
711 T obj = 0;
712 T objQ = clhs.quantileObjectiveFunc(sampleCountPerQuantile);
713 T objC = clhs.correlationObjectiveFunc(corr);
714
715 obj = objQ + objC;
716
717 //begin annealing schedule. If we have a perfect latin hypercube -- or if we pass enough iterations -- stop iterating.
718 while (temp > 0 && objQ != 0) {
719 uint64_t swpIndex; //the index of the sample
720 int i; //the index within the indices, x, y, and features vector so we know what to swap without searching
721 if (dist(rng) < 0.5) {
722 //50% of the time, choose a random sample to replace
723 i = indexDist(rng);
724 swpIndex = indices[i];
725 }
726 else {
727 //50% of the time, choose the worst sample to replace
728 int f, q, max;
729
730 //get feature and quantile to remove
731 for (f = 0; f < nFeat; f++) {
732 max = 0;
733
734 for (int s = 0; s < nSamp; s++) {
735 int count = sampleCountPerQuantile[f][s];
736
737 if (count > max) {
738 max = count;
739 q = s;
740 }
741 }
742
743 if (max != 1) {
744 break;
745 }
746 }
747
748 swpIndex = *(samplesPerQuantile[f][q].begin());
749 i = indicesMap.find(swpIndex)->second;
750 }
751
752 std::vector<T> oldf(nFeat);
753 for (int j = 0; j < nFeat; j++) {
754 oldf[j] = features[i * nFeat + j];
755 }
756
757 Point<T> p;
758 uint64_t newIndex = clhs.randomIndex();
759 while (indicesMap.contains(newIndex)) {
760 newIndex = clhs.randomIndex();
761 }
762 clhs.getPoint(p, newIndex);
763 for (int j = 0; j < nFeat; j++) {
764 features[i * nFeat + j] = p.p_features[j];
765 }
766
767 //recalculate sample count per quantile
768 std::vector<int> oldq(nFeat);
769 std::vector<int> newq(nFeat);
770 for (int f = 0; f < nFeat; f++) {
771 int q = getQuantile(oldf[f], quantiles[f]);
772 oldq[f] = q;
773 sampleCountPerQuantile[f][q]--;
774
775 q = getQuantile(p.p_features[f], quantiles[f]);
776 newq[f] = q;
777 sampleCountPerQuantile[f][q]++;
778 }
779
780 //recalculate objective function from quantiles
781 T newObjQ = clhs.quantileObjectiveFunc(sampleCountPerQuantile);
782
783 //recalculate correlation matrix
784 const auto result = oneapi::dal::compute(cor_desc, table); // we update the table in place
785 oneapi::dal::row_accessor<const T> acc {result.get_cor_matrix()};
786 for (int j = 0; j < nFeat; j++) {
787 auto row = acc.pull({j, j + 1});
788
789 for (int k = 0; k < nFeat; k++) {
790 corr[j][k] = row[k];
791 }
792 }
793
794 //recalculate objective function from correlation matrix
795 T newObjC = clhs.correlationObjectiveFunc(corr);
796
797 T newObj = newObjQ + newObjC;
798 T delta = newObj - obj;
799
800 bool keep = dist(rng) < std::exp(-1 * delta / temp);
801
802 if (keep) {
803 //update the new changes
804 x[i] = p.x;
805 y[i] = p.y;
806 indicesMap.erase(indices[i]);
807 indicesMap.emplace(newIndex, i);
808
809 for (int f = 0; f < nFeat; f++) {
810 samplesPerQuantile[f][oldq[f]].erase(indices[i]);
811 samplesPerQuantile[f][newq[f]].insert(newIndex);
812 }
813
814 indices[i] = newIndex;
815
816 objC = newObjC;
817 objQ = newObjQ;
818 obj = newObj;
819 }
820 else {
821 //revert anything already changed
822 for (int f = 0; f < nFeat; f++) {
823 sampleCountPerQuantile[f][newq[f]]--;
824 sampleCountPerQuantile[f][oldq[f]]++;
825 }
826
827 std::memcpy(
828 reinterpret_cast<void *>(features.data() + (i * nFeat)),
829 reinterpret_cast<void *>(oldf.data()),
830 nFeat * sizeof(T)
831 );
832 }
833
834 //update annealing temperature
835 temp -= d;
836 }
837
838 //add samples to output layer
839 for (int i = 0 ; i < nSamp; i++) {
840 double xCoord = GT[0] + x[i] * GT[1] + y[i] * GT[2];
841 double yCoord = GT[3] + x[i] * GT[4] + y[i] * GT[5];
842 OGRPoint point = OGRPoint(xCoord, yCoord);
843 helper::addPoint(&point, p_layer);
844
845 if (plot) {
846 xCoords.push_back(xCoord);
847 yCoords.push_back(yCoord);
848 }
849 }
850}

881std::tuple<std::vector<std::vector<double>>, vector::GDALVectorWrapper *>
882clhs(
883 raster::GDALRasterWrapper *p_raster,
884 int nSamp,
885 int iterations,
886 vector::GDALVectorWrapper *p_access,
887 std::string layerName,
888 double buffInner,
889 double buffOuter,
890 bool plot,
891 std::string tempFolder,
892 std::string filename)
893{
894 GDALAllRegister();
895
896 int width = p_raster->getWidth();
897 int height = p_raster->getHeight();
898 int nFeat = p_raster->getBandCount();
899 double *GT = p_raster->getGeotransform();
900
901 std::vector<double> xCoords, yCoords;
902
903 std::vector<helper::RasterBandMetaData> bands(p_raster->getBandCount());
904 for (int i = 0; i < nFeat; i++) {
905 bands[i].p_band = p_raster->getRasterBand(i);
906 bands[i].type = p_raster->getRasterBandType(i);
907 bands[i].size = p_raster->getRasterBandTypeSize(i);
908 bands[i].nan = bands[i].p_band->GetNoDataValue();
909 bands[i].p_band->GetBlockSize(&bands[i].xBlockSize, &bands[i].yBlockSize);
910 }
911
912 //create output dataset before doing anything which will take a long time in case of failure.
913 GDALDriver *p_driver = GetGDALDriverManager()->GetDriverByName("MEM");
914 if (!p_driver) {
915 throw std::runtime_error("unable to create output sample dataset driver.");
916 }
917 GDALDataset *p_samples = p_driver->Create("", 0, 0, 0, GDT_Unknown, nullptr);
918 if (!p_samples) {
919 throw std::runtime_error("unable to create output dataset with driver.");
920 }
921
922 vector::GDALVectorWrapper *p_wrapper = new vector::GDALVectorWrapper(p_samples, std::string(p_raster->getDataset()->GetProjectionRef()));
923 OGRLayer *p_layer = p_samples->CreateLayer("samples", p_wrapper->getSRS(), wkbPoint, nullptr);
924 if (!p_layer) {
925 throw std::runtime_error("unable to create output dataset layer.");
926 }
927
928 access::Access access(
929 p_access,
930 p_raster,
931 layerName,
932 buffInner,
933 buffOuter,
934 true,
935 tempFolder,
936 bands[0].xBlockSize,
937 bands[0].yBlockSize
938 );
939
940 //fast random number generator using xoshiro256++
941 //https://vigna.di.unimi.it/ftp/papers/ScrambledLinear.pdf
942 xso::xoshiro_4x64_plus rng;
943 uint64_t multiplier = helper::getProbabilityMultiplier(
944 width,
945 height,
946 p_raster->getPixelWidth(),
947 p_raster->getPixelHeight(),
948 4,
949 MILLION * 100,
950 false,
951 access.area
952 );
953 helper::RandValController rand(bands[0].xBlockSize, bands[0].yBlockSize, multiplier, &rng);
954
955 //get data type for all bands
956 GDALDataType type = GDT_Float32;
957 for (const helper::RasterBandMetaData& band : bands) {
958 if (band.type == GDT_Float64) {
959 type = GDT_Float64;
960 break;
961 }
962 }
963
964 if (type == GDT_Float64) {
965 std::vector<std::vector<double>> quantiles;
