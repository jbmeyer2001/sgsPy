56void calcSPQuantiles(
57 raster::GDALRasterWrapper *p_raster,
58 helper::RasterBandMetaData& band,
59 std::vector<double>& probabilities,
60 std::vector<double>& quantiles)
61{
62 std::vector<float> spProbabilities(probabilities.size());
63 std::vector<float> spQuantiles(quantiles.size());
64
65 //get float vector of probabilities
66 for (size_t i = 0; i < probabilities.size(); i++) {
67 spProbabilities[i] = static_cast<float>(probabilities[i]);
68 }
69
70 //filter out nan values
71 size_t fi = 0;
72 float nan = static_cast<float>(band.nan);
73 size_t pixelCount = static_cast<size_t>(p_raster->getWidth()) * static_cast<size_t>(p_raster->getHeight());
74 std::vector<float> filteredData(pixelCount);
75 for (size_t i = 0; i < pixelCount; i++) {
76 float val = helper::getPixelValueDependingOnType<float>(band.type, band.p_buffer, i);
77 bool isNan = std::isnan(val) || val == nan;
78 if (!isNan) {
79 filteredData[fi] = val;
80 fi ++;
81 }
82 }
83 filteredData.resize(fi + 1);
84
85 //define variables to pass to MKL quantiles calculation function
86 //https://www.intel.com/content/www/us/en/docs/onemkl/developer-reference-c/2025-2/vslsseditquantiles.html
87 VSLSSTaskPtr task;
88 int status;
89 MKL_INT quant_order_n = spQuantiles.size();
90 MKL_INT p = 1;
91 MKL_INT nparams = filteredData.size();
92 MKL_INT xstorage = VSL_SS_MATRIX_STORAGE_ROWS;
93 float *quant_order = spProbabilities.data();
94 float *params = filteredData.data();
95 float *quants = spQuantiles.data();
96
97 //calculate quantiles using MKL
98 status = vslsSSNewTask(&task, &p, &nparams, &xstorage, params, nullptr, nullptr);
99 status = vslsSSEditQuantiles(task, &quant_order_n, quant_order, quants, nullptr, nullptr);
100 status = vslsSSCompute(task, VSL_SS_QUANTS, VSL_SS_METHOD_FAST);
101 status = vslSSDeleteTask(&task);
102
103 //assign double quantile values
104 for(size_t i = 0; i < spQuantiles.size(); i++) {
105 quantiles[i] = static_cast<double>(spQuantiles[i]);
106 }
107}

236void batchCalcSPQuantiles(
237 raster::GDALRasterWrapper *p_raster,
238 helper::RasterBandMetaData& band,
239 std::vector<double>& probabilities,
240 std::vector<double>& quantiles,
241 std::mutex& mutex,
242 std::condition_variable& cv,
243 bool& calculated,
244 double eps)
245{
246 std::vector<float> spProbabilities(probabilities.size());
247 std::vector<float> spQuantiles(quantiles.size());
248
249 //get float vector of probabilities
250 for (size_t i = 0; i < probabilities.size(); i++) {
251 spProbabilities[i] = static_cast<float>(probabilities[i]);
252 }
253
254 int xBlocks = (p_raster->getWidth() + band.xBlockSize - 1) / band.xBlockSize;
255 int yBlocks = (p_raster->getHeight() + band.yBlockSize - 1) / band.yBlockSize;
256
257 float nan = static_cast<float>(band.nan);
258 float spEps = static_cast<float>(eps);
259 void *p_buffer = VSIMalloc3(band.xBlockSize, band.yBlockSize, band.size);
260 float *p_filtered = reinterpret_cast<float *>(VSIMalloc3(band.xBlockSize, band.yBlockSize, sizeof(float)));
261
262 //define variables to pass to MKL quantiles calculation function
263 //https://www.intel.com/content/www/us/en/docs/onemkl/developer-reference-c/2025-2/vslsseditstreamquantiles.html
264 VSLSSTaskPtr task;
265 int status;
266 MKL_INT quant_order_n = spProbabilities.size();
267 MKL_INT p = 1;
268 MKL_INT n = band.xBlockSize * band.yBlockSize;
269 MKL_INT nparams = VSL_SS_SQUANTS_ZW_PARAMS_N;
270 MKL_INT xstorage = VSL_SS_MATRIX_STORAGE_ROWS;
271
272 float *quant_order = reinterpret_cast<float *>(spProbabilities.data());
273 float *quants = reinterpret_cast<float *>(spQuantiles.data());
274
275 status = vslsSSNewTask(&task, &p, &n, &xstorage, p_filtered, nullptr, nullptr);
276
277 //read and compute raster by blocks
278 bool calledEditStreamQuantiles = false;
279 for (int yBlock = 0; yBlock < yBlocks; yBlock++) {
280 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
281 int xValid, yValid;
282 band.p_mutex->lock();
283 band.p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
284 CPLErr err = band.p_band->ReadBlock(xBlock, yBlock, p_buffer);
285 band.p_mutex->unlock();
286 if (err) {
287 throw std::runtime_error("error reading block from band.");
288 }
289
290 int fi = 0;
291 for (int y = 0; y < yValid; y++) {
292 int index = y * band.xBlockSize;
293 for (int x = 0; x < xValid; x++) {
294 float val = helper::getPixelValueDependingOnType<float>(band.type, p_buffer, index);
295 bool isNan = std::isnan(val) || val == nan;
296 if (!isNan) {
297 p_filtered[fi] = val;
298 fi++;
299 }
300 index++;
301 }
302 }
303
304 if (fi == 0) {
305 continue;
306 }
307
308 n = fi;
309 if (!calledEditStreamQuantiles) {
310 //for some reason this has issues if called before the first band of data is loaded,
311 //although it only has to be called once.
312 status = vslsSSEditStreamQuantiles(task, &quant_order_n, quant_order, quants, &nparams, &spEps);
313 calledEditStreamQuantiles = true;
314 }
315 status = vslsSSCompute(task, VSL_SS_STREAM_QUANTS, VSL_SS_METHOD_SQUANTS_ZW_FAST);
316 }
317 }
318
319 //use VSL_SS_METHOD_SQUANTS_ZW (not VSL_SS_METHOD_SQUANTS_ZW_FAST) to get final estimates
320 n = 0;
321 vslsSSCompute(task, VSL_SS_STREAM_QUANTS, VSL_SS_METHOD_SQUANTS_ZW);
322 status = vslSSDeleteTask(&task);
323
324 //assign double quantile values
325 for(size_t i = 0; i < spQuantiles.size(); i++) {
326 quantiles[i] = static_cast<double>(spQuantiles[i]);
327 }
328
329 VSIFree(p_buffer);
330 VSIFree(p_filtered);
331
332 mutex.lock();
333 calculated = true;
334 mutex.unlock();
335 cv.notify_all();
336}

514inline void processMapPixel(
515 size_t index,
516 helper::RasterBandMetaData& dataBand,
517 void * p_dataBuffer,
518 helper::RasterBandMetaData& stratBand,
519 void * p_stratBuffer,
520 std::vector<double>& quantiles,
521 size_t multiplier,
522 bool& mapNan,
523 size_t& mapStrat)
524{
525 double val = helper::getPixelValueDependingOnType<double>(dataBand.type, p_dataBuffer, index);
526 bool isNan = std::isnan(val) || (double)val == dataBand.nan;
527 mapNan |= isNan;
528
529 size_t strat = 0;
530 if (!isNan) {
531 auto it = std::lower_bound(quantiles.begin(), quantiles.end(), val);
532 strat = (it == quantiles.end()) ? quantiles.size() : std::distance(quantiles.begin(), it);
533 }
534
535 helper::setStrataPixelDependingOnType(stratBand.type, p_stratBuffer, index, isNan, strat);
536
537 if (!mapNan) {
538 mapStrat += strat * multiplier;
539 }
540}

712raster::GDALRasterWrapper *quantiles(
713 raster::GDALRasterWrapper *p_raster,
714 std::map<int, std::vector<double>> userProbabilites,
715 bool map,
716 std::string filename,
717 std::string tempFolder,
718 bool largeRaster,
719 int threadCount,
720 std::map<std::string, std::string> driverOptions,
721 double eps)
722{
723 GDALAllRegister();
724
725 int height = p_raster->getHeight();
726 int width = p_raster->getWidth();
727 double *geotransform = p_raster->getGeotransform();
728 std::string projection = std::string(p_raster->getDataset()->GetProjectionRef());
729
730 GDALDataset *p_dataset = nullptr;
731
732 int bandCount = userProbabilites.size();
733 std::vector<std::vector<double>> probabilities;
734 std::vector<std::string> bandNames = p_raster->getBands();
735
736 std::vector<helper::RasterBandMetaData> dataBands(bandCount);
737 std::vector<helper::RasterBandMetaData> stratBands(bandCount + map);
738 std::vector<helper::VRTBandDatasetInfo> VRTBandInfo;
739
740 bool isMEMDataset = !largeRaster && filename == "";
741 bool isVRTDataset = largeRaster && filename == "";
742
743 std::mutex dataBandMutex;
744 std::mutex stratBandMutex;
745 std::vector<std::mutex> stratBandMutexes(isVRTDataset * (bandCount + map));
746
747 std::string driver;
748 if (isMEMDataset || isVRTDataset) {
749 driver = isMEMDataset ? "MEM" : "VRT";
750 p_dataset = helper::createVirtualDataset(driver, width, height, geotransform, projection);
751 }
752 else {
753 std::filesystem::path filepath = filename;
754 std::string extension = filepath.extension().string();
755
756 if (extension == ".tif") {
757 driver = "Gtiff";
758 }
759 else {
760 throw std::runtime_error("sgs only supports .tif files right now");
761 }
762 }
763
764 //allocate, read, and initialize raster data and breaks information
765 GDALDataType stratPixelType = GDT_Int8;
766 size_t stratPixelSize = 1;
767 size_t band = 0;
768 for (auto const& [key, val] : userProbabilites) {
769 helper::RasterBandMetaData *p_dataBand = &dataBands[band];
770 helper::RasterBandMetaData *p_stratBand = &stratBands[band];
771
772 //get and store metadata from input raster band
773 GDALRasterBand *p_band = p_raster->getRasterBand(key);
774 p_dataBand->p_band = p_band;
775 p_dataBand->type = p_raster->getRasterBandType(key);
776 p_dataBand->size = p_raster->getRasterBandTypeSize(key);
777 p_dataBand->p_buffer = largeRaster ? nullptr : p_raster->getRasterBandBuffer(key);
778 p_dataBand->nan = p_band->GetNoDataValue();
779 p_dataBand->p_mutex = &dataBandMutex;
780 p_band->GetBlockSize(&p_dataBand->xBlockSize, &p_dataBand->yBlockSize);
781
782 //add probabilities
783 probabilities.push_back(val);
784
785 //update metadata of new strat raster
786 size_t maxStrata = val.size() + 1;
787 helper::setStratBandTypeAndSize(maxStrata, &p_stratBand->type, &p_stratBand->size);
788 p_stratBand->name = "strat_" + bandNames[key];
789 p_stratBand->xBlockSize = map ? dataBands[0].xBlockSize : p_dataBand->xBlockSize;
790 p_stratBand->yBlockSize = map ? dataBands[0].yBlockSize : p_dataBand->yBlockSize;
791 p_stratBand->p_mutex = isVRTDataset ? &stratBandMutexes[band] : &stratBandMutex;
792
793 //update dataset with new band information
794 if (isMEMDataset) {
795 helper::addBandToMEMDataset(p_dataset, *p_stratBand);
796 }
797 else if (isVRTDataset) {
798 helper::createVRTBandDataset(p_dataset, *p_stratBand, tempFolder, std::to_string(key), VRTBandInfo, driverOptions);
799 }
800 else {
801 if (stratPixelSize < p_stratBand->size) {
802 stratPixelSize = p_stratBand->size;
803 stratPixelType = p_stratBand->type;
804 }
805 }
806
807 band++;
808 }
809
810 //set multipliers if mapped stratification
811 std::vector<size_t> multipliers(probabilities.size(), 1);
812 if (map) {
813 //determine the stratification band index multiplier for the mapped band
814 for (int i = 0; i < bandCount - 1; i++) {
815 multipliers[i + 1] = multipliers[i] * (probabilities[i].size() + 1);
816 }
817
818 //update info of new strat raster band map
819 helper::RasterBandMetaData *p_stratBand = &stratBands.back();
820 size_t maxStrata = multipliers.back() * (probabilities.back().size() + 1);
821 helper::setStratBandTypeAndSize(maxStrata, &p_stratBand->type, &p_stratBand->size);
822 p_stratBand->name = "strat_map";
823 p_stratBand->xBlockSize = dataBands[0].xBlockSize;
824 p_stratBand->yBlockSize = dataBands[0].yBlockSize;
825 p_stratBand->p_mutex = isVRTDataset ? &stratBandMutexes.back() : &stratBandMutex;
826
827 //update dataset with band information
828 if (isMEMDataset) {
829 helper::addBandToMEMDataset(p_dataset, *p_stratBand);
830 }
831 else if (isVRTDataset) {
832 helper::createVRTBandDataset(p_dataset, *p_stratBand, tempFolder, "map", VRTBandInfo, driverOptions);
833 }
834 else {
835 if (stratPixelSize < p_stratBand->size) {
836 stratPixelSize = p_stratBand->size;
837 stratPixelType = p_stratBand->type;
838 }
839 }
840 }
841
842 //create full non-virtual dataset now that we have all required band information
843 if (!isMEMDataset && !isVRTDataset) {
844 bool useTiles = stratBands[0].xBlockSize != width &&
845 stratBands[0].yBlockSize != height;
846
847 for (size_t band = 0; band < stratBands.size(); band++) {
848 stratBands[band].size = stratPixelSize;
849 stratBands[band].type = stratPixelType;
850 stratBands[band].p_buffer = !largeRaster ? VSIMalloc3(height, width, stratPixelSize) : nullptr;
851 }
852
853 p_dataset = helper::createDataset(
854 filename,
855 driver,
856 width,
857 height,
858 geotransform,
859 projection,
860 stratBands.data(),
861 stratBands.size(),
862 useTiles,
863 driverOptions
864 );
865 }
866
867 std::vector<std::vector<double>> quantiles(probabilities.size());
868 if (largeRaster) {
869 pybind11::gil_scoped_acquire acquire;
870 boost::asio::thread_pool pool(threadCount);
871
872 //initialize synchronization variables
873 std::vector<std::mutex> mutexes(map ? 1 : bandCount);
874 std::vector<std::condition_variable> cvs(map ? 1 : bandCount);
875 bool *quantilesCalculated = reinterpret_cast<bool *>(VSIMalloc2(bandCount, sizeof(bool)));
876
877 //call batch processing quantiles function depending on data type
878 for (int i = 0; i < bandCount; i++) {
879 helper::RasterBandMetaData band = dataBands[i];
880 quantiles[i].resize(probabilities[i].size());
881 quantilesCalculated[i] = false;
882 if (band.type != GDT_Float64) {
883 boost::asio::post(pool, std::bind(batchCalcSPQuantiles,
884 p_raster,
885 std::ref(band),
886 std::ref(probabilities[i]),
887 std::ref(quantiles[i]),
888 std::ref(mutexes[map ? 0 : i]),
889 std::ref(cvs[map ? 0 : i]),
890 std::ref(quantilesCalculated[i]),
891 static_cast<double>(eps)
892 ));
893 }
894 else {
895 boost::asio::post(pool, std::bind(batchCalcDPQuantiles,
896 p_raster,
897 std::ref(band),
898 std::ref(probabilities[i]),
899 std::ref(quantiles[i]),
900 std::ref(mutexes[i]),
901 std::ref(cvs[i]),
902 std::ref(quantilesCalculated[i]),
903 static_cast<double>(eps)
904 ));
905 }
906 }
907
908 //iterate through all pixels and update the stratified raster bands
909 if (map) {
910
911 //NOTE: all processing threads in the case of a mapped raster must wait
912 //until quantiles have been calculated for every band.
913 std::unique_lock lock(mutexes[0]);
914 bool allBandsQuantilesCalculated = true;
915 for (int i = 0; i < bandCount; i++) {
916 allBandsQuantilesCalculated &= quantilesCalculated[i];
917 }
918
919 while (!allBandsQuantilesCalculated) {
920 cvs[0].wait(lock);
921 allBandsQuantilesCalculated = true;
922 for (int i = 0; i < bandCount; i++) {
923 allBandsQuantilesCalculated &= quantilesCalculated[i];
924 }
925 }
926
927 //use the first raster band to determine block size
928 int xBlockSize = dataBands[0].xBlockSize;
929 int yBlockSize = dataBands[0].yBlockSize;
930
931 int xBlocks = (p_raster->getWidth() + xBlockSize - 1) / xBlockSize;
932 int yBlocks = (p_raster->getHeight() + yBlockSize - 1) / yBlockSize;
933 int chunkSize = yBlocks / threadCount;
934
935 for (int yBlockStart = 0; yBlockStart < yBlocks; yBlockStart += chunkSize) {
936 int yBlockEnd = std::min(yBlockStart + chunkSize, yBlocks);
937
938 boost::asio::post(pool, [
939 bandCount,
940 xBlockSize,
941 yBlockSize,
942 yBlockStart,
943 yBlockEnd,
944 xBlocks,
945 &dataBands,
946 &stratBands,
947 &quantiles,
948 &multipliers
949 ] {
950 std::vector<void *> dataBuffers(dataBands.size());
951 std::vector<void *> stratBuffers(stratBands.size());
952 for (size_t band = 0; band < dataBuffers.size(); band++) {
953 dataBuffers[band] = VSIMalloc3(xBlockSize, yBlockSize, dataBands[band].size);
954 stratBuffers[band] = VSIMalloc3(xBlockSize, yBlockSize, stratBands[band].size);
955 }
956 stratBuffers.back() = VSIMalloc3(xBlockSize, yBlockSize, stratBands.back().size);
957
958 for (int yBlock = yBlockStart; yBlock < yBlockEnd; yBlock++) {
959 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
960 int xValid, yValid;
961 dataBands[0].p_mutex->lock();
962 dataBands[0].p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
963 dataBands[0].p_mutex->unlock();
964
965 //read raster band data into band buffers
966 for (size_t band = 0; band < static_cast<size_t>(bandCount); band++) {
