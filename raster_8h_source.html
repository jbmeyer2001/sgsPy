97 void readRasterBand(int width, int height, int band) {
98 GDALDataType type = this->getRasterBandType(band);
99 size_t size = this->getRasterBandTypeSize(band);
100 size_t max = std::numeric_limits<size_t>::max();
101
102 //perform size checks
103 if (max / size < static_cast<size_t>(width) ||
104 max / (size * static_cast<size_t>(width)) < static_cast<size_t>(height)) {
105 throw std::runtime_error("raster too large to fit in memory.");
106 }
107
108 if (static_cast<size_t>(height) * static_cast<size_t>(width) * size > GIGABYTE) {
109 throw std::runtime_error("sgs does not allow allocation of a raster into memory for direct pixel access purposes if it would be larger than 1 gigabyte.");
110 }
111
112 //allocate data
113 void *p_data = VSIMalloc3(height, width, size);
114
115 //perform raster read on current band
116 CPLErr err = this->p_dataset->GetRasterBand(band + 1)->RasterIO(
117 GF_Read, //GDALRWFlag eRWFlag
118 0, //int nXOff
119 0, //int nYOff
120 this->getWidth(), //int nXSize
121 this->getHeight(), //int nYSize
122 p_data, //void *pData
123 width, //int nBufXSize
124 height, //int nBufYSize
125 type, //GDALDataType eBufType
126 0, //int nPixelSpace
127 0 //int nLineSpace
128 );
129 if (err) {
130 throw std::runtime_error("error reading raster band from dataset.");
131 }
132
133 //update dislpay information as required
134 if (width != this->getWidth() || height != this->getHeight()) {
135 this->displayRasterBandRead[band] = true;
136 this->displayRasterBandPointers[band] = p_data;
137 }
138 else {
139 this->rasterBandRead[band] = true;
140 this->rasterBandPointers[band] = p_data;
141 }
142 }

172 void createFromDataset(GDALDataset *p_dataset) {
173 this->p_dataset = GDALDatasetUniquePtr(p_dataset);
174
175 //geotransform
176 CPLErr cplerr = this->p_dataset->GetGeoTransform(this->geotransform);
177 if (cplerr) {
178 throw std::runtime_error("error getting geotransform from dataset.");
179 }
180
181 //crs
182 this->crs = std::string(OGRSpatialReference(this->p_dataset->GetProjectionRef()).GetName());
183
184 //proj
185 char *p_proj;
186 OGRErr ogrerr = OGRSpatialReference(this->p_dataset->GetProjectionRef()).exportToPrettyWkt(&p_proj);
187 if (ogrerr) {
188 throw std::runtime_error("error getting projection as WKT from dataset.");
189 }
190 this->proj = std::string(p_proj);
191 CPLFree(p_proj);
192
193
194 //initialize (but don't read) raster band pointers
195 this->rasterBandPointers = std::vector<void *>(this->getBandCount(), nullptr);
196 this->rasterBandRead = std::vector<bool>(this->getBandCount(), false);
197 this->displayRasterBandPointers = std::vector<void *>(this->getBandCount(), nullptr);
198 this->displayRasterBandRead = std::vector<bool>(this->getBandCount(), false);

272 GDALRasterWrapper(py::buffer buffer, std::vector<double> geotransform, std::string projection, std::vector<double> nanVals, std::vector<std::string> names, std::string projDBPath) {
273 //set proj.db search path to search for the proj.db file which is included in sgs package
274 char **paths = nullptr;
275 paths = CSLAddString(paths, projDBPath.c_str());
276 OSRSetPROJSearchPaths(paths);
277 CSLDestroy(paths);
278
279 py::buffer_info info = buffer.request();
280
281 //get height width and band count from pybuffer
282 int width, height;
283 size_t bandCount, bandSize;
284 if (info.ndim == 3) {
285 bandCount = info.shape[0];
286 height = info.shape[1];
287 width = info.shape[2];
288 bandSize = info.strides[0];
289 }
290 else if (info.ndim == 2) {
291 bandCount = 1;
292 height = info.shape[0];
293 width = info.shape[1];
294 }
295 else {
296 throw std::runtime_error("dimension of numpy array must be 2 or 3");
297 }
298
299 if (bandCount != names.size()) {
300 throw std::runtime_error("band names array does not have the same number of bands as the py buffer");
301 }
302
303 //get data type from pybuffer
304 GDALDataType type;
305 size_t size;
306 if (info.format == py::format_descriptor<int8_t>::format()) {
307 type = GDT_Int8;
308 size = sizeof(int8_t);
309 }
310 else if (info.format == py::format_descriptor<int16_t>::format()) {
311 type = GDT_Int16;
312 size = sizeof(int16_t);
313 }
314 else if (info.format == py::format_descriptor<uint16_t>::format()) {
315 type = GDT_UInt16;
316 size = sizeof(uint16_t);
317 }
318 else if (info.format == py::format_descriptor<int32_t>::format()) {
319 type = GDT_Int32;
320 size = sizeof(int32_t);
321 }
322 else if (info.format == py::format_descriptor<uint32_t>::format()) {
323 type = GDT_UInt32;
324 size = sizeof(uint32_t);
325 }
326 else if (info.format == py::format_descriptor<float>::format()) {
327 type = GDT_Float32;
328 size = sizeof(float);
329 }
330 else if (info.format == py::format_descriptor<double>::format()) {
331 type = GDT_Float64;
332 size = sizeof(double);
333 }
334 else {
335 throw std::runtime_error("data type of array must be one of int8, int16, uint16, int32, uint32, float32, or float64.");
336 }
337
338 GDALAllRegister();
339 GDALDataset *p_dataset = helper::createVirtualDataset("MEM", width, height, geotransform.data(), projection);
340 std::vector<void *> bands(bandCount);
341 for (size_t i = 0; i < bandCount; i++) {
342 helper::RasterBandMetaData band;
343 band.p_buffer = (void *)((size_t)info.ptr + (i * bandSize));
344 band.type = type;
345 band.size = size;
346 band.nan = nanVals[i];
347 band.name = names[i];
348 helper::addBandToMEMDataset(p_dataset, band);
349 bands[i] = band.p_buffer;
350
351 }
352
353 this->createFromDataset(p_dataset);
354 this->rasterBandPointers = bands;
355 this->rasterBandRead = std::vector<bool>(bandCount, true);
356 this->externalRasterData = true;
357 }

622 py::buffer getRasterBandAsMemView(int width, int height, int band) {
623 bool display = (width != this->getWidth() || height != this->getHeight());
624 void *p_buffer;
625 GDALDataType type = this->getRasterBandType(band);
626
627 //allocate raster if required
628 if (!display && !this->rasterBandRead[band]) {
629 this->readRasterBand(width, height, band);
630 }
631
632 //(re)allocate display raster if required
633 if (display) {
634 if (width != this->displayRasterWidth || height != this->displayRasterHeight) {
635 free(this->displayRasterBandPointers[band]);
636 this->displayRasterBandRead[band] = false;
637 }
638
639 if (this->displayRasterBandRead[band] == false) {
640 this->readRasterBand(width, height, band);
641 }
642 }
643
644 //get the (allocated) data buffer
645 p_buffer = (!display) ?
646 this->rasterBandPointers[band] :
647 this->displayRasterBandPointers[band];
648
649 switch(type) {
650 case GDT_Int8:
651 return getBuffer<int8_t>(sizeof(int8_t), p_buffer, width, height);
652 case GDT_UInt16:
653 return getBuffer<uint16_t>(sizeof(uint16_t), p_buffer, width, height);
654 case GDT_Int16:
655 return getBuffer<int16_t>(sizeof(int16_t), p_buffer, width, height);
656 case GDT_UInt32:
657 return getBuffer<uint32_t>(sizeof(uint32_t), p_buffer, width, height);
658 case GDT_Int32:
659 return getBuffer<int32_t>(sizeof(int32_t), p_buffer, width, height);
660 case GDT_Float32:
661 return getBuffer<float>(sizeof(float), p_buffer, width, height);
662 case GDT_Float64:
663 return getBuffer<double>(sizeof(double), p_buffer, width, height);
664 default:
665 throw std::runtime_error("raster pixel data type not supported.");
666 }
667 }
