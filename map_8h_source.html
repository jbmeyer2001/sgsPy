121 int height = rasters[0]->getHeight();
122 int width = rasters[0]->getWidth();
123 double *geotransform = rasters[0]->getGeotransform();
124 std::string projection = (rasters[0]->getDataset()->GetProjectionRef());
125 if (projection == "") {
126 throw std::runtime_error("could not get projection from the first raster argument.");
127 }
128
129 OGRSpatialReference srs;
130 srs.importFromWkt(projection.c_str());
131
132 for (size_t i = 1; i < rasters.size(); i++) {
133 if (rasters[i]->getHeight() != height) {
134 std::string err = "raster with index " + std::to_string(i) + " has a different height from the raster at index 0.";
135 throw std::runtime_error(err);
136 }
137
138 if (rasters[i]->getWidth() != width) {
139 std::string err = "raster with index " + std::to_string(i) + " has a different width from the raster at index 0.";
140 throw std::runtime_error(err);
141 }
142
143 double *checkGeotransform = rasters[i]->getGeotransform();
144 for (int j = 0; j < 6; j++) {
145 if (geotransform[i] != checkGeotransform[i]) {
146 std::string err = "raster with index " + std::to_string(i) + " has a different geotransform from the raster at index 0.";
147 throw std::runtime_error(err);
148 }
149 }
150
151 OGRSpatialReference checkSrs;
152 checkSrs.importFromWkt(rasters[i]->getDataset()->GetProjectionRef());
153 if (!srs.IsSame(&checkSrs)) {
154 std::string err = "raster with index " + std::to_string(i) + " has a different projection from the raster at index 0.";
155 throw std::runtime_error(err);
156 }
157 }

159 std::vector<helper::RasterBandMetaData> stratBands;
160 helper::RasterBandMetaData mapBand;
161 std::vector<helper::VRTBandDatasetInfo> VRTBandInfo(1);
162
163 bool isMEMDataset = !largeRaster && filename == "";
164 bool isVRTDataset = largeRaster && filename == "";
165
166 std::vector<std::mutex> stratDatasetMutexes(rasters.size());
167 std::mutex mapBandMutex;
168
169 std::string driver;
170 GDALDataset *p_dataset = nullptr;
171 if (isMEMDataset || isVRTDataset) {
172 std::string driver = isMEMDataset ? "MEM" : "VRT";
173 p_dataset = helper::createVirtualDataset(driver, width, height, geotransform, projection);
174 }
175 else {
176 std::filesystem::path filepath = filename;
177 std::string extension = filepath.extension().string();
178
179 if (extension == ".tif") {
180 driver = "Gtiff";
181 }
182 else {
183 throw std::runtime_error("sgs only supports .tif files right now");
184 }
185 }
186
187 //step 1 iterate through bands populating rasterBands and bandStratMultiplier objects
188 std::vector<size_t> multipliers(1, 1);
189 for (size_t i = 0; i < rasters.size(); i++) {
190 raster::GDALRasterWrapper *p_raster = rasters[i];
191
192 for (size_t j = 0; j < bands[i].size(); j++) {
193 stratBands.resize(stratBands.size() + 1);
194 helper::RasterBandMetaData *p_stratBand = &stratBands.back();
195
196 int band = bands[i][j];
197 int strataCount = strataCounts[i][j];
198
199 GDALRasterBand *p_band = p_raster->getRasterBand(band);
200 p_stratBand->p_band = p_band;
201 p_stratBand->type = p_raster->getRasterBandType(band);
202 p_stratBand->size = p_raster->getRasterBandTypeSize(band);
203 p_stratBand->p_buffer = largeRaster ? nullptr : p_raster->getRasterBandBuffer(band);
204 p_stratBand->nan = p_band->GetNoDataValue();
205 p_stratBand->p_mutex = &stratDatasetMutexes[i];
206 p_band->GetBlockSize(&p_stratBand->xBlockSize, &p_stratBand->yBlockSize);
207
208 helper::printTypeWarningsForInt32Conversion(p_stratBand->type);
209 multipliers.push_back(multipliers.back() * strataCount);
210 }
211 }
212
213 size_t bandCount = stratBands.size();
214 size_t maxStrata = multipliers.back();
215 multipliers.pop_back();
216 helper::setStratBandTypeAndSize(maxStrata, &mapBand.type, &mapBand.size);
217 mapBand.name = "strat_map";
218 mapBand.xBlockSize = stratBands[0].xBlockSize;
219 mapBand.yBlockSize = stratBands[0].yBlockSize;
220 mapBand.p_mutex = &mapBandMutex;

222 if (isMEMDataset) {
223 helper::addBandToMEMDataset(p_dataset, mapBand);
224 }
225 else if (isVRTDataset) {
226 helper::createVRTBandDataset(
227 p_dataset,
228 mapBand,
229 tempFolder,
230 "map",
231 VRTBandInfo,
232 driverOptions
233 );
234 }
235 else {
236 bool useTiles = mapBand.xBlockSize != width &&
237 mapBand.yBlockSize != height;
238
239 mapBand.p_buffer = !largeRaster ?
240 VSIMalloc3(height, width, mapBand.size) :
241 nullptr;
242
243 p_dataset = helper::createDataset(
244 filename,
245 driver,
246 width,
247 height,
248 geotransform,
249 projection,
250 &mapBand,
251 1,
252 useTiles,
253 driverOptions
254 );
255 }
256
257 if (largeRaster) {
258 pybind11::gil_scoped_acquire acquire;
259 boost::asio::thread_pool pool(threadCount);
260
261 int xBlockSize = stratBands[0].xBlockSize;
262 int yBlockSize = stratBands[0].yBlockSize;
263
264 int xBlocks = (width + xBlockSize - 1) / xBlockSize;
265 int yBlocks = (height + yBlockSize - 1) / yBlockSize;
266 int chunkSize = yBlocks / threadCount;
267
268 for (int yBlockStart = 0; yBlockStart < yBlocks; yBlockStart += chunkSize) {
269 int yBlockEnd = std::min(yBlockStart + chunkSize, yBlocks);
270
271 boost::asio::post(pool, [
272 bandCount,
273 xBlockSize,
274 yBlockSize,
275 yBlockStart,
276 yBlockEnd,
277 xBlocks,
278 &stratBands,
279 &mapBand,
280 &multipliers
281 ] {
282 std::vector<void *> stratBuffers(bandCount);
283 for (size_t band = 0; band < bandCount; band++) {
284 stratBuffers[band] = VSIMalloc3(xBlockSize, yBlockSize, stratBands[band].size);
285 }
286 void *p_mapBuffer = VSIMalloc3(xBlockSize, yBlockSize, mapBand.size);
287
288 //cast to int so there is less data type conversion while processign
289 std::vector<int> intNoDataValues(bandCount);
290 for (size_t band = 0; band < bandCount; band++) {
291 intNoDataValues[band] = static_cast<int>(stratBands[band].nan);
292 }
293
294 for (int yBlock = yBlockStart; yBlock < yBlockEnd; yBlock++) {
295 for (int xBlock = 0; xBlock < xBlocks; xBlock++) {
296 int xValid, yValid;
297 stratBands[0].p_mutex->lock();
298 stratBands[0].p_band->GetActualBlockSize(xBlock, yBlock, &xValid, &yValid);
299 stratBands[0].p_mutex->unlock();
300
301 for (size_t band = 0; band < bandCount; band++) {
302 rasterBandIO(
303 stratBands[band],
304 stratBuffers[band],
305 xBlockSize,
306 yBlockSize,
307 xBlock,
308 yBlock,
309 xValid,
310 yValid,
311 true //read = true
312 );
313 }
314
315 for (int y = 0; y < yValid; y++) {
316 for (int x = 0; x < xValid; x++) {
317 size_t index = static_cast<size_t>(x + y * xBlockSize);
318 bool isNan = false;
319 int mappedStrat = 0;
320
321 for (size_t band = 0; band < bandCount; band++) {
322 int strat = helper::getPixelValueDependingOnType<int>(stratBands[band].type, stratBuffers[band], index);
323 isNan = strat == intNoDataValues[band];
324
325 if (!isNan) {
326 mappedStrat += strat * multipliers[band];
327 }
328 else {
329 break;
330 }
331 }
332
333 helper::setStrataPixelDependingOnType(mapBand.type, p_mapBuffer, index, isNan, mappedStrat);
334 }
335 }
336
337 helper::rasterBandIO(
338 mapBand,
339 p_mapBuffer,
340 xBlockSize,
341 yBlockSize,
342 xBlock,
343 yBlock,
344 xValid,
345 yValid,
346 false //read = false
347 );
348 }
349 }
350
351 for (size_t band = 0; band < bandCount; band++) {
352 VSIFree(stratBuffers[band]);
353 }
354 VSIFree(p_mapBuffer);
355 });
356 }
357
358 pool.join();
359 pybind11::gil_scoped_release release;
360 }
361 else {
362 std::vector<int> intNoDataValues(bandCount);
363 for (size_t band = 0; band < bandCount; band++) {
364 intNoDataValues[band] = static_cast<int>(stratBands[band].nan);
365 }
366
367 size_t pixelCount = static_cast<size_t>(height) * static_cast<size_t>(width);
368 for (size_t index = 0; index < pixelCount; index++) {
369 bool isNan = false;
370 int mappedStrat = 0;
371
372 for (size_t band = 0; (band < bandCount) && !isNan; band++) {
373 int strat = helper::getPixelValueDependingOnType<int>(stratBands[band].type, stratBands[band].p_buffer, index);
374 isNan = strat == intNoDataValues[band];
375 if (!isNan) {
376 mappedStrat += strat * multipliers[band];
377 }
378 }
379
380 helper::setStrataPixelDependingOnType(mapBand.type, mapBand.p_buffer, index, isNan, mappedStrat);
381 }
382
383 if (!isVRTDataset && !isMEMDataset) {
384 CPLErr err = mapBand.p_band->RasterIO(
385 GF_Write,
386 0,
387 0,
388 width,
389 height,
390 mapBand.p_buffer,
391 width,
392 height,
393 mapBand.type,
394 0,
395 0
396 );
397 if (err) {
398 throw std::runtime_error("error writing band to file.");
399 }
400 }
401 }
402
403 //close and add VRT sub dataset as band
404 if (isVRTDataset) {
405 GDALClose(VRTBandInfo[0].p_dataset);
406 helper::addBandToVRTDataset(p_dataset, mapBand, VRTBandInfo[0]);
407 }
408
409 return largeRaster ?
410 new raster::GDALRasterWrapper(p_dataset) :
411 new raster::GDALRasterWrapper(p_dataset, {mapBand.p_buffer});
412}
